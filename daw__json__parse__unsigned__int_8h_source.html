41  template<typename Signed, typename Unsigned>
42  constexpr Signed to_signed( Unsigned &&u, Signed sign ) {
43  using unsigned_t = daw::remove_cvref_t<Unsigned>;
44  if constexpr( sizeof( Signed ) >= sizeof( intmax_t ) and
45  daw::is_system_integral_v<unsigned_t> and
46  daw::is_system_integral_v<Signed> and
47  sizeof( Signed ) == sizeof( unsigned_t ) ) {
48  if( DAW_UNLIKELY( u == ( static_cast<unsigned_t>(
49  ( daw::numeric_limits<Signed>::max )( ) ) +
50  1 ) ) ) {
51  // the bits of static_cast<unsigned_t>( limits<Signed>::max( ) ) + 1
52  // are the same as limits<Signed>::min( ). We can just cast
53  return static_cast<Signed>( u );
54  }
55  return static_cast<Signed>( sign * static_cast<Signed>( u ) );
56  } else {
57  return static_cast<Signed>( sign * static_cast<Signed>( u ) );
58  }

61  [[nodiscard]] static inline constexpr bool
62  is_made_of_eight_digits_cx( const char *ptr ) {
63  // The copy to local buffer is to get the compiler to treat it like a
64  // reinterpret_cast
65 
66  std::byte const buff[8]{
67  static_cast<std::byte>( ptr[0] ), static_cast<std::byte>( ptr[1] ),
68  static_cast<std::byte>( ptr[2] ), static_cast<std::byte>( ptr[3] ),
69  static_cast<std::byte>( ptr[4] ), static_cast<std::byte>( ptr[5] ),
70  static_cast<std::byte>( ptr[6] ), static_cast<std::byte>( ptr[7] ) };
71 
72  UInt64 val = UInt64( );
73  for( std::size_t n = 0; n < 8; ++n ) {
74  val |= to_uint64( buff[n] ) << ( 8 * n );
75  }
76  return ( ( ( val & 0xF0F0'F0F0'F0F0'F0F0_u64 ) |
77  ( ( ( val + 0x0606'0606'0606'0606_u64 ) &
78  0xF0F0'F0F0'F0F0'F0F0_u64 ) >>
79  4U ) ) == 0x3333'3333'3333'3333_u64 );
80  }

90  // Constexpr'ified version from
91  // https://kholdstare.github.io/technical/2020/05/26/faster-integer-parsing.html
92  inline constexpr UInt64 parse_8_digits( const char *const str ) {
93  auto const chunk = daw::to_uint64_buffer( str );
94  // 1-byte mask trick (works on 4 pairs of single digits)
95  auto const lower_digits =
96  ( chunk & 0x0F'00'0F'00'0F'00'0F'00_u64 ) >> 8U;
97  auto const upper_digits =
98  ( chunk & 0x00'0F'00'0F'00'0F'00'0F_u64 ) * 10U;
99  auto const chunk2 = lower_digits + upper_digits;
100 
101  // 2-byte mask trick (works on 2 pairs of two digits)
102  auto const lower_digits2 =
103  ( chunk2 & 0x00'FF'00'00'00'FF'00'00_u64 ) >> 16U;
104  auto const upper_digits2 =
105  ( chunk2 & 0x00'00'00'FF'00'00'00'FF_u64 ) * 100U;
106  auto const chunk3 = lower_digits2 + upper_digits2;
107 
108  // 4-byte mask trick (works on pair of four digits)
109  auto const lower_digits3 =
110  ( chunk3 & 0x00'00'FF'FF'00'00'00'00_u64 ) >> 32U;
111  auto const upper_digits3 =
112  ( chunk3 & 0x00'00'00'00'00'00'FF'FF_u64 ) * 10000U;
113  auto const chunk4 = lower_digits3 + upper_digits3;
114 
115  return chunk4 & 0xFFFF'FFFF_u64;
116  }

118  "8 digit parser does not work on this platform" );
119  inline constexpr UInt64 parse_16_digits( const char *const str ) {
120  auto const upper = parse_8_digits( str );
121  auto const lower = parse_8_digits( str + 8 );
122  return upper * 100'000'000_u64 + lower;
123  }
124  static_assert( parse_16_digits( "1234567890123456" ) ==
125  1234567890123456_u64,
126  "16 digit parser does not work on this platform" );

132  unsigned_parser( constexpr_exec_tag, ParseState &parse_state ) {

135  using result_t = max_unsigned_t<RangeChecked, Unsigned, UInt64>;
136  using uresult_t =
137  max_unsigned_t<RangeChecked, daw::make_unsigned_t<Unsigned>, UInt64>;
138  static_assert(
139  not static_cast<bool>( RangeChecked ) or
140  std::is_same_v<uresult_t, UInt64>,
141  "Range checking is only supported for std integral types" );
142 
143  CharT *first = parse_state.first;
144  CharT *const last = parse_state.last;
145  uresult_t result = uresult_t( );
146 
147  while( last - first >= 16 ) {
148  result *= static_cast<uresult_t>( 10'000'000'000'000'000ULL );
149  result += static_cast<uresult_t>( parse_16_digits( first ) );
150  first += 16;
151  }
152  if( last - first >= 8 ) {
153  result *= static_cast<uresult_t>( 100'000'000ULL );
154  result += static_cast<uresult_t>( parse_8_digits( first ) );
155  first += 8;
156  }
157  if constexpr( ParseState::is_zero_terminated_string( ) ) {
158  auto dig = parse_digit( *first );
159  while( dig < 10U ) {
160  result *= 10U;
161  result += dig;
162  ++first;
163  dig = parse_digit( *first );
164  }
165  } else {
166  while( first < last ) {
167  result *= 10U;
168  result += parse_digit( *first );
169  ++first;
170  }
171  }
172  if constexpr( RangeChecked != options::JsonRangeCheck::Never ) {
173  auto const count = ( daw::numeric_limits<result_t>::digits10 + 1U ) -
174  std::size( parse_state );
175  daw_json_ensure(
176  ( ( result <= static_cast<uresult_t>(
177  ( daw::numeric_limits<result_t>::max )( ) ) ) &
178  ( count >= 0 ) ),
179  ErrorReason::NumberOutOfRange, parse_state );
180  }
181  parse_state.first = first;
182  if constexpr( RangeChecked == options::JsonRangeCheck::Never ) {
183  return daw::construct_a<Unsigned>( static_cast<Unsigned>( result ) );
184  } else {
185  return daw::construct_a<Unsigned>(
186  daw::narrow_cast<Unsigned>( result ) );
187  }
188  }
