66  // Paths are specified with dot separators, if the name has a dot in it,
67  // it must be escaped
68  // memberA.memberB.member\.C has 3 parts['memberA', 'memberB', 'member.C']
69  [[nodiscard]] static constexpr pop_json_path_result
70  pop_json_path( daw::string_view &path ) {
71  auto result = pop_json_path_result{ };
72  if( path.empty( ) ) {
73  return result;
74  }
75  if( path.front( ) == '.' ) {
76  path.remove_prefix( );
77  }
78  result.current =
79  path.pop_front_until( [&, in_escape = false]( char c ) mutable {
80  if( in_escape ) {
81  in_escape = false;
82  return false;
83  }
84  switch( c ) {
85  case '\\':
86  in_escape = true;
87  return false;
88  case '.':
89  case '[':
90  case ']':
91  result.found_char = c;
92  return true;
93  default:
94  return false;
95  }
96  } );
97  return result;
98  }

100  [[nodiscard]] static constexpr bool
101  json_path_compare( daw::string_view json_path_item,
102  daw::string_view member_name ) {
103  if( json_path_item.front( ) == '\\' ) {
104  json_path_item.remove_prefix( );
105  }
106  while( not json_path_item.empty( ) and not member_name.empty( ) ) {
107  if( json_path_item.front( ) != member_name.front( ) ) {
108  return false;
109  }
110  json_path_item.remove_prefix( );
111  if( not json_path_item.empty( ) and
112  json_path_item.front( ) == '\\' ) {
113  json_path_item.remove_prefix( );
114  }

116  }
117  return std::size( json_path_item ) == std::size( member_name );

135  daw::string_view path ) {
136 
137  auto pop_result = pop_json_path( path );

139  if( pop_result.found_char == ']' ) {
140  // Array Index

142  ErrorReason::InvalidJSONPath, parse_state );

146  pop_result.current );
147

150  (void)skip_value( parse_state );
151  parse_state.trim_left_checked( );
152  if( ( idx > 0 ) & ( parse_state.has_more( ) and
153  ( parse_state.front( ) != ',' ) ) ) {
154  return false;
155  }
156  parse_state.move_next_member_or_end( );
157  }
158  } else {
159  daw_json_assert_weak( parse_state.is_opening_brace_checked( ),
160  ErrorReason::InvalidJSONPath, parse_state );
161  parse_state.remove_prefix( );
162  parse_state.trim_left_unchecked( );
163  auto name = parse_name( parse_state );
164  while( not json_path_compare( pop_result.current, name ) ) {
