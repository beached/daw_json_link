412  template<bool is_unchecked_input>
413  DAW_ATTRIBUTE_FLATTEN [[nodiscard]] static inline constexpr char const *
414  skip_digits( char const *first, char const *const last ) {
415  if( DAW_JSON_UNLIKELY( first >= last ) ) {
416  return first;
417  }
418  unsigned dig = parse_digit( *first );
419  while( dig < 10 ) {
420  ++first;
421  if constexpr( not is_unchecked_input ) {
422  if( DAW_JSON_UNLIKELY( first >= last ) ) {
423  break;
424  }
425  }
426  dig = parse_digit( *first );
427  }
428  return first;
429  }
430  /***
431  * Skip a number and store the position of it's components in the returned
432  * Range
433  */
434  template<typename Range>
435  [[nodiscard]] static constexpr Range skip_number( Range &rng ) {
436  auto result = rng;
437  char const *first = rng.first;
438  char const *const last = rng.last;
439  daw_json_assert_weak( first < last, ErrorReason::UnexpectedEndOfData, rng );
440  if( *first == '-' ) {
441  ++first;
442  }
443  first = skip_digits<Range::is_unchecked_input>( first, last );
444  char const *decimal = nullptr;
445  if( ( Range::is_unchecked_input or first < last ) and ( *first == '.' ) ) {
446  decimal = first;
447  ++first;
448  first = skip_digits<Range::is_unchecked_input>( first, last );
449  }
450  char const *exp = nullptr;
451  unsigned dig = parse_digit( *first );
452  if( ( Range::is_unchecked_input or ( first < last ) ) &
453  ( ( dig == parsed_constants::e_char ) |
454  ( dig == parsed_constants::E_char ) ) ) {
455  exp = first;
456  ++first;
457  daw_json_assert_weak( first < last, ErrorReason::UnexpectedEndOfData,
458  [&] {
459  auto r = rng;
460  r.first = first;
461  return r;
462  }( ) );
463  dig = parse_digit( *first );
464  if( ( dig == parsed_constants::plus_char ) |
465  ( dig == parsed_constants::minus_char ) ) {
466  ++first;
467  }
468  first = skip_digits<Range::is_unchecked_input>( first, last );
469  }
470 
471  rng.first = first;
472  result.last = first;
473  result.class_first = decimal;
474  result.class_last = exp;
475  return result;
476  }

478  /***
479  * When we don't know ahead of time what we are skipping switch on the first
480  * value and call that types specific skipper
481  * TODO: Investigate if there is a difference for the times we know what the
482  * member should be if that can increase performance
483  */
484  template<typename Range>
485  [[nodiscard]] static inline constexpr Range skip_value( Range &rng ) {
486  daw_json_assert_weak( rng.has_more( ), ErrorReason::UnexpectedEndOfData,
487  rng );
488 
489  // reset counter
490  rng.counter = 0;
491  switch( rng.front( ) ) {
492  case '"':
493  return skip_string( rng );
494  case '[':
495  return rng.skip_array( );
496  case '{':
497  return rng.skip_class( );
498  case 't':
499  return skip_true( rng );
500  case 'f':
501  return skip_false( rng );
502  case 'n':
503  return skip_null( rng );
504  case '-':
505  case '0':
506  case '1':
507  case '2':
508  case '3':
509  case '4':
510  case '5':
511  case '6':
512  case '7':
513  case '8':
514  case '9':
515  return skip_number( rng );
516  }
517  if constexpr( Range::is_unchecked_input ) {
518  DAW_UNREACHABLE( );
519  } else {
520  daw_json_error( ErrorReason::InvalidStartOfValue, rng );
521  }
522  }
