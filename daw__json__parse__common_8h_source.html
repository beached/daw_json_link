59  [[maybe_unused]] DAW_ATTRIB_FLATINLINE static inline constexpr auto
60  construct_value( template_params<Value, Constructor>,
61  ParseState &parse_state, Args &&...args ) {
62  // Silence MSVC warning, used in other if constexpr case
63  (void)parse_state;
64  if constexpr( ParseState::has_allocator ) {
65  auto alloc = parse_state.get_allocator_for( template_arg<Value> );
66  return daw::try_alloc_construct<Value, Constructor>(
67  DAW_MOVE( alloc ), DAW_FWD( args )... );
68  } else {
69  static_assert(
70  std::is_invocable_v<Constructor, Args...>,
71  "Unable to construct value with the supplied arguments" );
72  return Constructor{ }( DAW_FWD2( Args, args )... );
73  }

108  template<typename Value, typename Constructor, typename ParseState,
109  typename... Args>
110  [[maybe_unused]] DAW_ATTRIB_FLATINLINE static inline constexpr auto
111  construct_value_tp( ParseState &parse_state,
112  fwd_pack<Args...> &&tp_args ) {
113  if constexpr( ParseState::has_allocator ) {
114  using alloc_t =
115  typename ParseState::template allocator_type_as<Value>;
116  auto alloc = parse_state.get_allocator_for( template_arg<Value> );
117  if constexpr( std::is_invocable_v<Constructor, Args..., alloc_t> ) {
118  return construct_value_tp_invoke<Constructor>(
119  DAW_MOVE( tp_args ), DAW_MOVE( alloc ),
120  std::index_sequence_for<Args...>{ } );
121  } else if constexpr( std::is_invocable_v<Constructor,
122  std::allocator_arg_t,
123  alloc_t, Args...> ) {
124  return construct_value_tp_invoke<Constructor>(
125  std::allocator_arg, DAW_MOVE( alloc ), DAW_MOVE( tp_args ),
126  std::index_sequence_for<Args...>{ } );
127  } else {
128  static_assert(
129  std::is_invocable_v<Constructor, Args...>,
130  "Unable to construct value with the supplied arguments" );
131  return construct_value_tp_invoke<Constructor>(
132  DAW_MOVE( tp_args ), std::index_sequence_for<Args...>{ } );
133  }
134  } else {
135  // Silence MSVC warning, used in other if constexpr case
136  (void)parse_state;
137  static_assert(
138  std::is_invocable_v<Constructor, Args...>,
139  "Unable to construct value with the supplied arguments" );
140  return construct_value_tp_invoke<Constructor>(
141  DAW_MOVE( tp_args ), std::index_sequence_for<Args...>{ } );
142  }
143  }
