60  template<typename Result, typename Unsigned>
61  DAW_ATTRIB_FLATINLINE static inline constexpr Result
62  power10( constexpr_exec_tag, Result result, Unsigned p ) {
63  // We only have a double table, of which float is a subset. Long double
64  // will be calculated in terms of that
65 
66  constexpr auto max_v =
67  static_cast<Result>( dpow10_tbl[max_exp<Result>] );
68 
69  if( DAW_UNLIKELY( p > max_exp<Result> ) ) {
70  if constexpr( std::is_floating_point_v<Result> ) {
71  if( DAW_UNLIKELY( p >
72  daw::numeric_limits<Result>::max_exponent ) ) {
73  return std::numeric_limits<Result>::infinity( );
74  }
75  }
76  Result exp2 = max_v;
77  p -= max_exp<Result>;
78 
79  for( ; p > max_exp<Result>; p -= max_exp<Result> ) {
80  exp2 *= max_v;
81  }
82  return static_cast<Result>( result ) *
83  ( exp2 * static_cast<Result>(
84  dpow10_tbl[static_cast<std::size_t>( p )] ) );
85  } else if( DAW_UNLIKELY( p < -max_exp<Result> ) ) {
86  if constexpr( std::is_floating_point_v<Result> ) {
87  if( DAW_UNLIKELY( p <
88  daw::numeric_limits<Result>::min_exponent ) ) {
89  return Result{ 0 };
90  }
91  }
92  Result exp2 = max_v;
93  p += max_exp<Result>;
94 
95  for( ; p < -max_exp<Result>; p += max_exp<Result> ) {
96  result /= max_v;
97  }
98  return ( static_cast<Result>( result ) /
99  static_cast<Result>(
100  dpow10_tbl[static_cast<std::size_t>( -p )] ) ) /
101  exp2;
102  }
103  if( p < 0 ) {
104  return static_cast<Result>( result ) /
105  static_cast<Result>(
106  dpow10_tbl[static_cast<std::size_t>( -p )] );
107  }
108  return static_cast<Result>( result ) *
109  static_cast<Result>( dpow10_tbl[static_cast<std::size_t>( p )] );
110  }

112  template<typename Result, typename Unsigned>
113  DAW_ATTRIB_FLATINLINE static inline constexpr Result
114  power10( runtime_exec_tag, Result result, Unsigned p ) {
115  if constexpr( std::is_same_v<Result, double> or
116  std::is_same_v<Result, float> ) {
117  return power10( constexpr_exec_tag{ }, result,
118  static_cast<std::int32_t>( p ) );
119  } else {
120  // For long double and others fallback to the slower std::pow
121  using std::pow;
122  return result * pow( static_cast<Result>( 10.0 ), p );
123  }
