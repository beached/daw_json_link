254  template<typename JsonElement = json_value, auto... ParsePolicies>
255  std::vector<json_lines_range<JsonElement, ParsePolicies...>>
256  partition_jsonl_document( std::size_t num_partitions,
257  daw::string_view jsonl_doc ) {
258  using result_t =
259  std::vector<json_lines_range<JsonElement, ParsePolicies...>>;
260  if( num_partitions <= 1 ) {
261  return result_t{
262  json_lines_range<JsonElement, ParsePolicies...>( jsonl_doc ) };
263  }
264  auto approx_segsize = jsonl_doc.size( ) / num_partitions;
265  auto result = result_t{ };
266  char const *const last = daw::data_end( jsonl_doc );
267  while( not jsonl_doc.empty( ) ) {
268  char const *tmp = std::data( jsonl_doc ) + approx_segsize;
269  if( tmp >= last ) {
270  result.emplace_back( jsonl_doc );
271  break;
272  }
273  while( tmp < last and *tmp != '\n' ) {
274  ++tmp;
275  }
276  if( tmp < last ) {
277  ++tmp;
278  }
279  auto sz = static_cast<std::size_t>( tmp - std::data( jsonl_doc ) );
280  auto doc = jsonl_doc.pop_front( sz );
281  doc.trim_suffix( );
282  if( not doc.empty( ) ) {
283  result.emplace_back( doc );
284  }
285  }
286  return result;
287  }
