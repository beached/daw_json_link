88  /***
89  * Move parser until member name matches key if needed
90  * @param member to move_to
91  * @return position of member or size
92  */
93  [[nodiscard]] constexpr std::size_t move_to( json_member_name member ) {
94  std::size_t pos = 0;
95  std::size_t const Sz = std::size( m_locs );
96  for( ; pos < Sz; ++pos ) {
97  if( m_locs[pos].is_match( member.name, member.hash_value ) ) {
98  return pos;
99  }
100  }
101 
102  auto it = [&] {
103  if( m_locs.empty( ) ) {
104  return m_value.begin( );
105  }
106  auto res = m_locs.back( ).location;
107  ++res;
108  return res;
109  }( );
110  auto const last = m_value.end( );
111  while( it != last ) {
112  auto name = it.name( );
113  daw_json_assert_weak( name, ErrorReason::MissingMemberName );
114  auto const &new_loc = m_locs.emplace_back(
115  daw::string_view( std::data( *name ), std::size( *name ) ), it );
116  if( new_loc.is_match( member.name ) ) {
117  return pos;
118  }
119  ++pos;
120  ++it;
121  }
122  return std::size( m_locs );
123  }
124 
125  /***
126  * Move parser until member index matches, if needed
127  * @param index position of member to move to
128  * @return position in members or size
129  */
130  [[nodiscard]] constexpr std::size_t move_to( std::size_t index ) {
131  if( index < std::size( m_locs ) ) {
132  return index;
133  }
134  auto it = [&] {
135  if( m_locs.empty( ) ) {
136  return m_value.begin( );
137  }
138  auto res = m_locs.back( ).location;
139  ++res;
140  return res;
141  }( );
142  auto last = m_value.end( );
143  std::size_t pos = std::size( m_locs );
144  while( it != last ) {
145  auto name = it.name( );
146  if( name ) {
147  m_locs.emplace_back(
148  daw::string_view( std::data( *name ), std::size( *name ) ), it );
149  } else {
150  m_locs.emplace_back( daw::string_view( ), it );
151  }
152  if( pos == index ) {
153  return pos;
154  }
155  ++pos;
156  ++it;
157  }
158  return std::size( m_locs );
159  }
