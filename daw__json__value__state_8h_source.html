73  /***
74  * Move parser until member name matches key if needed
75  * @param member to move_to
76  * @return position of member or size
77  */
78  [[nodiscard]] constexpr std::size_t move_to( json_member_name member ) {
79  std::size_t pos = 0;
80  std::size_t const Sz = std::size( m_locs );
81  for( ; pos < Sz; ++pos ) {
82  if( m_locs[pos].is_match( member.name, member.hash_value ) ) {
83  return pos;
84  }
85  }
86 
87  auto it = [&] {
88  if( m_locs.empty( ) ) {
89  return m_value.begin( );
90  }
91  auto res = m_locs.back( ).location;
92  ++res;
93  return res;
94  }( );
95  auto const last = m_value.end( );
96  while( it != last ) {
97  auto name = it.name( );
98  daw_json_assert_weak( name, ErrorReason::MissingMemberName );
99  auto const &new_loc = m_locs.emplace_back(
100  daw::string_view( std::data( *name ), std::size( *name ) ), it );
101  if( new_loc.is_match( member.name ) ) {
102  return pos;
103  }
104  ++pos;
105  ++it;
106  }
107  return std::size( m_locs );
108  }
109 
110  /***
111  * Move parser until member index matches, if needed
112  * @param index position of member to move to
113  * @return position in members or size
114  */
115  [[nodiscard]] constexpr std::size_t move_to( std::size_t index ) {
116  if( index < std::size( m_locs ) ) {
117  return index;
118  }
119  auto it = [&] {
120  if( m_locs.empty( ) ) {
121  return m_value.begin( );
122  }
123  auto res = m_locs.back( ).location;
124  ++res;
125  return res;
126  }( );
127  auto last = m_value.end( );
128  std::size_t pos = std::size( m_locs );
129  while( it != last ) {
130  auto name = it.name( );
131  if( name ) {
132  m_locs.emplace_back(
133  daw::string_view( std::data( *name ), std::size( *name ) ), it );
134  } else {
135  m_locs.emplace_back( daw::string_view( ), it );
136  }
137  if( pos == index ) {
138  return pos;
139  }
140  ++pos;
141  ++it;
142  }
143  return std::size( m_locs );
144  }
