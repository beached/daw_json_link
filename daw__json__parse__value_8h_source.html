72  [[nodiscard]] DAW_ATTRIB_INLINE static constexpr json_result<JsonMember>
73  parse_value_real( ParseState &parse_state ) {
74  using constructor_t = typename JsonMember::constructor_t;
75  using element_t = typename JsonMember::base_type;
76 
77  if constexpr( JsonMember::literal_as_string !=
78  options::LiteralAsStringOpt::Never ) {
79  if constexpr( not KnownBounds ) {
80  skip_quote_when_literal_as_string<JsonMember::literal_as_string>(
81  parse_state );
82  }
83  if constexpr( JsonMember::allow_number_errors ==
84  options::JsonNumberErrors::AllowInf or
85  JsonMember::allow_number_errors ==
86  options::JsonNumberErrors::AllowNanInf ) {
87  element_t sign = element_t( 1.0 );
88  if( parse_state.front( ) == '-' ) {
89  sign = element_t( -1.0 );
90  parse_state.first++;
91  }
92  // Looking for Inf as that will match Infinity too.
93  if( parse_state.size( ) >= 4 and
94  parse_state.starts_with( "Inf" ) ) {
95 
96  parse_state.first += 3;
97  if( parse_state.front( ) == '"' ) {
98  parse_state.first++;
99  } else if( parse_state.size( ) >= 6 and
100  parse_state.starts_with( R"(inity")" ) ) {
101  parse_state.first += 6;
102  } else {
103  daw_json_error( ErrorReason::InvalidString, parse_state );
104  }
105  if constexpr( KnownBounds ) {
106  daw_json_assert_weak( parse_state.empty( ),
107  ErrorReason::InvalidNumber, parse_state );
108  } else {
109  daw_json_assert_weak(
110  parse_state.empty( ) or parse_policy_details::at_end_of_item(
111  parse_state.front( ) ),
112  ErrorReason::InvalidEndOfValue, parse_state );
113  }
114  return daw::cxmath::copy_sign(
115  daw::numeric_limits<element_t>::infinity( ), sign );
116  } else if( sign < element_t( 0 ) ) {
117  parse_state.first--;
118  }
119  }
120  if constexpr( JsonMember::allow_number_errors ==
121  options::JsonNumberErrors::AllowNaN or
122  JsonMember::allow_number_errors ==
123  options::JsonNumberErrors::AllowNanInf ) {
124  if( parse_state.starts_with( "NaN" ) ) {
125  parse_state.template move_to_next_of<'"'>( );
126  parse_state.remove_prefix( );
127  if constexpr( KnownBounds ) {
128  daw_json_assert_weak( parse_state.empty( ),
129  ErrorReason::InvalidNumber, parse_state );
130  } else {
131  daw_json_assert_weak(
132  parse_state.empty( ) or parse_policy_details::at_end_of_item(
133  parse_state.front( ) ),
134  ErrorReason::InvalidEndOfValue, parse_state );
135  }
136  return daw::numeric_limits<element_t>::quiet_NaN( );
137  }
138  }
139  }
140  if constexpr( KnownBounds and JsonMember::literal_as_string ==
141  options::LiteralAsStringOpt::Never ) {
142  return construct_value(
143  template_args<json_result<JsonMember>, constructor_t>, parse_state,
144  parse_real<element_t, true>( parse_state ) );
145  } else {
146  daw_json_assert_weak(
147  parse_state.has_more( ) and
148  parse_policy_details::is_number_start( parse_state.front( ) ),
149  ErrorReason::InvalidNumberStart, parse_state );
150 
151  auto result = construct_value(
152  template_args<json_result<JsonMember>, constructor_t>, parse_state,
153  parse_real<element_t, false>( parse_state ) );
154 
155  if constexpr( KnownBounds ) {
156  daw_json_assert_weak(
157  parse_state.empty( ) or
158  parse_policy_details::at_end_of_item( parse_state.front( ) ),
159  ErrorReason::InvalidEndOfValue, parse_state );
160  } else {
161  if constexpr( JsonMember::literal_as_string !=
162  options::LiteralAsStringOpt::Never ) {
163  skip_quote_when_literal_as_string<JsonMember::literal_as_string>(
164  parse_state );
165  }
166  daw_json_assert_weak(
167  parse_state.empty( ) or
168  parse_policy_details::at_end_of_item( parse_state.front( ) ),
169  ErrorReason::InvalidEndOfValue, parse_state );
170  }
171  return result;
172  }

363  parse_value_bool( ParseState &parse_state ) {
364  using constructor_t = typename JsonMember::constructor_t;
365 
366  if constexpr( KnownBounds ) {
367  // We have already checked if it is a true/false
368  if constexpr( ParseState::is_unchecked_input ) {
369  return static_cast<bool>( parse_state.counter );
370  } else {
371  switch( parse_state.front( ) ) {
372  case 't':
373  return construct_value(
374  template_args<json_result<JsonMember>, constructor_t>,
375  parse_state, true );
376  case 'f':
377  return construct_value(
378  template_args<json_result<JsonMember>, constructor_t>,
379  parse_state, false );
380  }
381  daw_json_error( ErrorReason::InvalidLiteral, parse_state );
382  }
383  } else {
384  // Beginning quotes

386  options::LiteralAsStringOpt::Never ) {
387  skip_quote_when_literal_as_string<JsonMember::literal_as_string>(
388  parse_state );
389  }

391  if constexpr( ParseState::is_unchecked_input ) {
392  if( parse_state.front( ) == 't' ) /* true */ {
393  result = true;
394  parse_state.remove_prefix( 4 );
395  } else /* false */ {
396  parse_state.remove_prefix( 5 );
397  }
398  } else {
399  if( parse_state.starts_with( "true" ) ) {
400  parse_state.remove_prefix( 4 );
401  result = true;
402  } else if( parse_state.starts_with( "false" ) ) {
403  parse_state.remove_prefix( 5 );
404  } else {
405  daw_json_error( ErrorReason::InvalidLiteral, parse_state );
406  }
407  }
408  // Trailing quotes
409  if constexpr( JsonMember::literal_as_string !=
410  options::LiteralAsStringOpt::Never ) {
411  skip_quote_when_literal_as_string<JsonMember::literal_as_string>(
412  parse_state );
413  }
414  parse_state.trim_left( );
415  daw_json_assert_weak(
416  not parse_state.has_more( ) or
417  parse_policy_details::at_end_of_item( parse_state.front( ) ),
418  ErrorReason::InvalidEndOfValue, parse_state );
419  return construct_value(
420  template_args<json_result<JsonMember>, constructor_t>, parse_state,
421  result );
