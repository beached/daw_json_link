231  [[nodiscard,
232  maybe_unused]] DAW_ONLY_INLINE inline constexpr json_result<JsonMember>
233  parse_value( ParseTag<JsonParseTypes::Bool>, Range &rng ) {
234  daw_json_assert_weak( std::size( rng ) >= 4, ErrorReason::InvalidLiteral,
235  rng );
236 
237  using constructor_t = typename JsonMember::constructor_t;
238 
239  if constexpr( KnownBounds ) {
240  // We have already checked if it is a true/false

243  } else {

245  case 't':

248  case 'f':

254  } else {

257  skip_quote_when_literal_as_string<JsonMember::literal_as_string>( rng );

261  if( rng.front( ) == 't' ) /* true */ {
262  result = true;
263  rng.remove_prefix( 4 );
264  } else /* false */ {
265  rng.remove_prefix( 5 );
266  }
267  } else {
268  if( rng.starts_with( "true" ) ) {
269  rng.remove_prefix( 4 );
270  result = true;
271  } else if( rng.starts_with( "false" ) ) {
272  rng.remove_prefix( 5 );
273  } else {
274  daw_json_error( ErrorReason::InvalidLiteral, rng );
275  }
276  }
277  if constexpr( JsonMember::literal_as_string !=
278  LiteralAsStringOpt::Never ) {
279  skip_quote_when_literal_as_string<JsonMember::literal_as_string>( rng );
280  }
281  rng.trim_left( );
282  daw_json_assert_weak(
283  parse_policy_details::at_end_of_item( rng.front( ) ),
284  ErrorReason::InvalidEndOfValue, rng );
285  return construct_value<json_result<JsonMember>>( constructor_t{ }, rng,

317  } else {

319
