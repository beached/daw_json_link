131  daw::string_view key ) const {
132  UInt32 const hash = name_hash<expect_long_strings>( key );
133 #if defined( DAW_JSON_BUGFIX_MSVC_EVAL_ORDER_002 )
134  (void)start_pos;

136 #else

138 #endif

197  /***
198  * Get the position from already seen JSON members or move the parser
199  * forward until we reach the end of the class or the member.
200  * @tparam N Number of members in json_class
201  * @tparam ParseState see IteratorRange
202  * @param locations members location and names
203  * @param parse_state Current JSON data
204  * @return IteratorRange with begin( ) being start of value
205  */
206  enum class AllMembersMustExist { yes, no };
207  template<std::size_t pos, AllMembersMustExist must_exist,
208  bool from_start = false, std::size_t N, typename ParseState,
209  bool B, typename CharT>
210  [[nodiscard]] DAW_ATTRIB_INLINE static constexpr std::pair<ParseState,
211  bool>
212  find_class_member( ParseState &parse_state,
213  locations_info_t<N, CharT, B> &locations,
214  bool is_nullable, daw::string_view member_name ) {
215 
216  // silencing gcc9 warning as these are selectively used
217  (void)is_nullable;
218  (void)member_name;
219 
220  daw_json_assert_weak(
221  nsc_or( is_nullable, ( not locations[pos].missing( ) ),
222  ( not parse_state.is_closing_brace_checked( ) ) ),
223  missing_member( member_name ), parse_state );
224 
225  parse_state.trim_left_unchecked( );
226  bool known = not locations[pos].missing( );
227  while( nsc_and( locations[pos].missing( ),
228  ( parse_state.front( ) != '}' ) ) ) {
229  // TODO: fully unescape name
230  // parse_name checks if we have more and are quotes
231  auto const name = parse_name( parse_state );
232  auto const name_pos =
233  locations.template find_name<ParseState::expect_long_strings>(
234  template_vals<( from_start ? 0 : pos )>, name );
235  if constexpr( must_exist == AllMembersMustExist::yes ) {
236  daw_json_assert_weak( name_pos < std::size( locations ),
237  ErrorReason::UnknownMember, parse_state );
238  } else {
239 #if defined( DAW_JSON_PARSER_DIAGNOSTICS )
240  std::cerr << "DEBUG: Unknown member '" << name << '\n';
241 #endif
242  if( name_pos >= std::size( locations ) ) {
243  // This is not a member we are concerned with
244  (void)skip_value( parse_state );
245  parse_state.move_next_member_or_end( );
246  continue;
247  }
248  }
249  if( name_pos == pos ) {
250  locations[pos].set_range( parse_state );
251  break;
252  } else {
253 #if defined( DAW_JSON_PARSER_DIAGNOSTICS )
254  std::cerr << "DEBUG: Out of order member '"
255  << locations.names[name_pos].name
256  << "' found, looking for '" << locations.names[pos].name
257  << ". It is "
258  << std::abs( static_cast<long long>( pos ) -
259  static_cast<long long>( name_pos ) )
260  << " members ahead in constructor\n";
261 #endif
262  // We are out of order, store position for later
263  // OLDTODO: use type knowledge to speed up skip
264  // OLDTODO: on skipped classes see if way to store
265  // member positions so that we don't have to
266  // re-parse them after
267  // RESULT: storing preparsed is slower, don't try 3 times
268  // it also limits the type of things we can parse potentially
269  // Using locations to switch on BaseType is slower too
270  locations[name_pos].set_range( skip_value( parse_state ) );
271 
272  if constexpr( ParseState::is_unchecked_input ) {
273  if( name_pos + 1 < std::size( locations ) ) {
274  parse_state.move_next_member( );
275  } else {
276  parse_state.move_next_member_or_end( );
277  }
278  } else {
279  parse_state.move_next_member_or_end( );
280  }
281  }
282  }
283  if( locations[pos].missing( ) ) {
284  known = true;
285  }
286  if constexpr( ParseState::has_allocator ) {
287  return std::pair<ParseState, bool>{
288  locations[pos]
289  .get_range( template_arg<ParseState> )
290  .with_allocator( parse_state ),
291  known };
292  } else {
293  return std::pair<ParseState, bool>{
294  locations[pos].get_range( template_arg<ParseState> ), known };
295  }
296  }
