55 /***
56 * Iterator for iterating over JSON array's
57 * @tparam JsonElement type under underlying element in array. If
58 * heterogeneous, a basic_json_value_iterator may be more appropriate
59 * @tparam ParsePolicy Parsing policy type
60 */
61 template<typename JsonElement, typename ParsePolicy = NoCommentSkippingPolicyChecked>
62 class json_array_iterator {
63 using CharT = typename ParsePolicy::CharT;
64
65 template<typename String>
66 static inline constexpr ParsePolicy get_range( String &&data, std::string_view member_path ) {
67 static_assert( std::is_convertible_v<decltype( std::data( data ) ), CharT *>,
68 "Attempt to assign a const char * to a char *" );
69
70 auto [is_found, result] = json_details::find_range<ParsePolicy>(
71 DAW_FWD2( String, data ),
72 { std::data( member_path ), std::size( member_path ) } );
73 daw_json_assert( is_found, ErrorReason::JSONPathNotFound );
74 daw_json_assert( result.front( ) == '[', ErrorReason::InvalidArrayStart, result );
75 return result;
76 }
77
78 public:
79 using element_type = typename json_details::json_deduced_type<JsonElement>;
80 static_assert( traits::not_same<element_type, void>::value, "Unknown JsonElement type." );
81 using value_type = typename element_type::parse_to_t;
82 using reference = value_type;
83 using pointer = json_details::arrow_proxy<value_type>;
84 using difference_type = std::ptrdiff_t;
85 // Can do forward iteration and be stored
86 using iterator_category = std::input_iterator_tag;
87
88 private:
89 using ParseState = ParsePolicy;
90 ParseState m_state = ParseState( );
91 /***
92 * This lets us fastpath and just skip n characters as we have already
93 * parsed them
94 */
95 mutable CharT *m_can_skip = nullptr;

97 public:
98 inline constexpr json_array_iterator( ) = default;

146 /***
147 * Parse the current element
148 * @pre good( ) returns true
149 * @return The parsed result of ParseElement
150 */
151 [[nodiscard]] inline constexpr value_type operator*( ) const {
152 daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
153 ErrorReason::UnexpectedEndOfData,
154 m_state );
155
156 auto tmp = m_state;
157
158 if constexpr( json_details::use_direct_construction_v<ParsePolicy, element_type> ) {
159 auto const run_after_parse = op_star_cleanup<CharT, ParseState>{ m_can_skip, tmp };
160 (void)run_after_parse;
161 return json_details::parse_value<element_type>(
162 tmp,
163 ParseTag<element_type::expected_type>{ } );
164 } else {
165 auto result =
166 json_details::parse_value<element_type>( tmp,
167 ParseTag<element_type::expected_type>{ } );
168
169 m_can_skip = tmp.first;
170 return result;
171 }
172 }
173
174 /***
175 * A dereferencable value proxy holding the result of operator*
176 * This is for compatibility with the Iterator concepts and should be
177 * avoided
178 * @pre good( ) returns true
179 * @return an arrow_proxy of the operator* result
180 */
181 [[nodiscard]] inline pointer operator->( ) const {
182 return pointer{ operator*( ) };
183 }
184
185 /***
186 * Move the parse state to the next element
187 * @return iterator after moving
188 */
189 inline constexpr json_array_iterator &operator++( ) {
190 daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
191 ErrorReason::UnexpectedEndOfData,
192 m_state );
193 if( m_can_skip ) {
194 m_state.first = m_can_skip;
195 m_can_skip = nullptr;
196 } else {
197 (void)json_details::skip_known_value<element_type>( m_state );
198 }
199 m_state.move_next_member_or_end( );
200 return *this;
201 }
202
203 /***
204 * Move the parse state to the next element
205 * @return iterator prior to moving
206 */
207 inline constexpr json_array_iterator operator++( int ) & {
208 auto tmp = *this;
209 (void)operator++( );
210 return tmp;
211 }

213 /***
214 * Is it ok to dereference iterator
215 * @return true when there is parse data available
216 */
217 [[nodiscard]] inline constexpr bool good( ) const {
218 return not m_state.is_null( ) and m_state.has_more( ) and m_state.front( ) != ']';
219 }
220
221 /***
222 * Are we good( )
223 * @return result of good( )
224 */
225 [[nodiscard]] explicit inline constexpr operator bool( ) const {
226 return good( );
227 }

229 /***
230 * Compare rhs for equivalence
231 * @param rhs Another json_array_iterator
232 * @return true when equivalent to rhs
233 */
234 [[nodiscard]] inline constexpr bool operator==( json_array_iterator const &rhs ) const {
235 if( not( *this ) ) {
236 return not rhs;
237 }
238 if( not rhs ) {
239 return false;
240 }
241 return ( m_state.first == rhs.m_state.first );
242 }
243
244 /***
245 * Check if the other iterator is not equivalent
246 * @param rhs another json_array_iterator
247 * @return true when rhs is not equivalent
248 */
249 [[nodiscard]] inline constexpr bool operator!=( json_array_iterator const &rhs ) const {
250 if( not( *this ) ) {
251 return static_cast<bool>( rhs );
252 }
253 if( not rhs ) {
254 return true;
255 }
256 return m_state.first != rhs.m_state.first;
257 }
