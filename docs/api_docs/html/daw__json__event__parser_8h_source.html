329 class DefaultJsonEventParserStackPolicy {
330 std::vector<StackValue> m_stack{ };

332 public:

336 using difference_type = std::ptrdiff_t;

341 m_stack.push_back( DAW_MOVE( v ) );
342 }

345 return m_stack.back( );

361 template<typename ParseState = NoCommentSkippingPolicyChecked,
362 typename StackContainerPolicy = DefaultJsonEventParserStackPolicy<
363 JsonEventParserStackValue<ParseState>>,
364 typename Handler>
365 inline constexpr void
366 json_event_parser( basic_json_value<ParseState> jvalue,
367 Handler &&handler ) {
368 using iterator = basic_json_value_iterator<ParseState>;
369 using json_value_t = basic_json_pair<ParseState>;
370 using stack_value_t = JsonEventParserStackValue<ParseState>;
371
372 auto parent_stack = StackContainerPolicy( );
373 long long class_depth = 0;
374 long long array_depth = 0;
375
376 auto const move_to_last = [&]( ) {
377 parent_stack.back( ).value.first = parent_stack.back( ).value.second;
378 };
379
380 auto const process_value = [&]( json_value_t p ) {
381 {
382 auto result = json_details::handle_on_value( handler, p );
383 switch( result.value ) {
384 case json_parse_handler_result::Complete:
385 parent_stack.clear( );
386 return;
387 case json_parse_handler_result::SkipClassArray:
388 move_to_last( );
389 return;
390 case json_parse_handler_result::Continue:
391 break;
392 }
393 }
394
395 auto &jv = p.value;
396 switch( jv.type( ) ) {
397 case JsonBaseParseTypes::Array: {
398 ++array_depth;
399 auto result = json_details::handle_on_array_start( handler, jv );
400 switch( result.value ) {
401 case json_parse_handler_result::Complete:
402 parent_stack.clear( );
403 return;
404 case json_parse_handler_result::SkipClassArray:
405 move_to_last( );
406 return;
407 case json_parse_handler_result::Continue:
408 break;
409 }
410 parent_stack.push_back(
411 { StackParseStateType::Array,
412 std::pair<iterator, iterator>( jv.begin( ), jv.end( ) ) } );
413 } break;
414 case JsonBaseParseTypes::Class: {
415 ++class_depth;
416 auto result = json_details::handle_on_class_start( handler, jv );
417 switch( result.value ) {
418 case json_parse_handler_result::Complete:
419 parent_stack.clear( );
420 return;
421 case json_parse_handler_result::SkipClassArray:
422 move_to_last( );
423 return;
424 case json_parse_handler_result::Continue:
425 break;
426 }
427 parent_stack.push_back(
428 { StackParseStateType::Class,
429 std::pair<iterator, iterator>( jv.begin( ), jv.end( ) ) } );
430 } break;
431 case JsonBaseParseTypes::Number: {
432 auto result = json_details::handle_on_number( handler, jv );
433 switch( result.value ) {
434 case json_parse_handler_result::Complete:
435 parent_stack.clear( );
436 return;
437 case json_parse_handler_result::SkipClassArray:
438 move_to_last( );
439 return;
440 case json_parse_handler_result::Continue:
441 break;
442 }
443 } break;
444 case JsonBaseParseTypes::Bool: {
445 auto result = json_details::handle_on_bool( handler, jv );
446 switch( result.value ) {
447 case json_parse_handler_result::Complete:
448 parent_stack.clear( );
449 return;
450 case json_parse_handler_result::SkipClassArray:
451 move_to_last( );
452 return;
453 case json_parse_handler_result::Continue:
454 break;
455 }
456 } break;
457 case JsonBaseParseTypes::String: {
458 auto result = json_details::handle_on_string( handler, jv );
459 switch( result.value ) {
460 case json_parse_handler_result::Complete:
461 parent_stack.clear( );
462 return;
463 case json_parse_handler_result::SkipClassArray:
464 move_to_last( );
465 return;
466 case json_parse_handler_result::Continue:
467 break;
468 }
469 } break;
470 case JsonBaseParseTypes::Null: {
471 auto result = json_details::handle_on_null( handler, jv );
472 switch( result.value ) {
473 case json_parse_handler_result::Complete:
474 parent_stack.clear( );
475 return;
476 case json_parse_handler_result::SkipClassArray:
477 move_to_last( );
478 return;
479 case json_parse_handler_result::Continue:
480 break;
481 }
482 } break;
483 case JsonBaseParseTypes::None:
484 default: {
485 auto result = json_details::handle_on_error( handler, jv );
486 switch( result.value ) {
487 case json_parse_handler_result::Complete:
488 parent_stack.clear( );
489 return;
490 case json_parse_handler_result::SkipClassArray:
491 move_to_last( );
492 return;
493 case json_parse_handler_result::Continue:
494 break;
495 }
496 } break;
497 }
498 };
499
500 auto const process_range = [&]( stack_value_t v ) {
501 if( v.value.first != v.value.second ) {
502 auto jv = *v.value.first;
503 v.value.first++;
504 parent_stack.push_back( DAW_MOVE( v ) );
505 process_value( DAW_MOVE( jv ) );
506 } else {
507 switch( v.type ) {
508 case StackParseStateType::Class: {
509 daw_json_assert_weak(
510 ( class_depth > 0 ) &
511 ( v.value.first.get_raw_state( ).has_more( ) and
512 v.value.first.get_raw_state( ).front( ) == '}' ),
513 ErrorReason::InvalidEndOfValue );
514 --class_depth;
515 auto result = json_details::handle_on_class_end( handler );
516 switch( result.value ) {
517 case json_parse_handler_result::Complete:
518 parent_stack.clear( );
519 return;
520 case json_parse_handler_result::SkipClassArray:
521 case json_parse_handler_result::Continue:
522 break;
523 }
524 } break;
525 case StackParseStateType::Array: {
526 daw_json_assert_weak(
527 ( array_depth > 0 ) &
528 ( v.value.first.get_raw_state( ).has_more( ) and
529 v.value.first.get_raw_state( ).front( ) == ']' ),
530 ErrorReason::InvalidEndOfValue );
531 --array_depth;
532 auto result = json_details::handle_on_array_end( handler );
533 switch( result.value ) {
534 case json_parse_handler_result::Complete:
535 parent_stack.clear( );
536 return;
537 case json_parse_handler_result::SkipClassArray:
538 case json_parse_handler_result::Continue:
539 break;
540 }
541 } break;
542 }
543 }
544 };
545
546 process_value( { std::nullopt, DAW_MOVE( jvalue ) } );
547
548 while( not parent_stack.empty( ) ) {
549 auto v = DAW_MOVE( parent_stack.back( ) );
550 parent_stack.pop_back( );
551 process_range( v );
552 }
553 daw_json_assert( class_depth == 0 and array_depth == 0,
554 ErrorReason::InvalidEndOfValue );
555 }
