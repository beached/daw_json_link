36 template<class Float, typename ParseState>
37 inline Float chars_to_fp( ParseState &parse_state ) {
38
39 using unsigned_t = typename fp_properties<Float>::unsigned_t;
40 using DECIMAL_POINT = daw::constant<'.'>;
41 using BASE = daw::constant<10U>;
42 using EXPONENT_MARKER = daw::constant<'e'>;
43 using MANTISSA_MAX =
44 daw::constant<unsigned_t{ 1U } << ( fp_properties<Float>::mantissa_width +
45 1U )>; // The extra bit is to give
46 // space for the implicit 1
47 using BITSTYPE_MAX_DIV_BY_BASE =
48 daw::constant<std::numeric_limits<unsigned_t>::max( ) / BASE::value>;
49 using CharT = typename ParseState::CharT;
50
51 CharT *const first_orig = parse_state.first;
52
53 auto result = fp_data<Float>{ };
54 auto cur_char = parse_state.safe_get( );
55 if( parse_state.safe_get( ) == '-' ) {
56 result.set_sign( true );
57 ++parse_state.first;
58 cur_char = parse_state.safe_get( );
59 }
60
61 bool seen_digit = false;
62 bool truncated = false;
63 if( fp_utils_details::is_digit( cur_char ) ) {
64 bool after_decimal = false;
65 CharT *number_start = parse_state.first;
66 unsigned_t mantissa = 0;
67 std::int32_t exponent = 0;
68
69 // The goal for this step of parsing is to
70 // convert the number in parse_state to the format mantissa * (base ^
71 // exponent)
72
73 // Before decimal point
74 auto digit = fp_utils_details::to_digit( cur_char );
75 if( digit < 10U ) {
76 seen_digit = true;
77 }
78 while( digit < 10U and mantissa < BITSTYPE_MAX_DIV_BY_BASE::value ) {
79 mantissa = ( mantissa * BASE::value ) + digit;
80 ++parse_state.first;
81 cur_char = parse_state.safe_get( );
82 digit = fp_utils_details::to_digit( cur_char );
83 }
84 if( cur_char == DECIMAL_POINT::value ) {
85 after_decimal = true;
86 ++parse_state.first;
87 cur_char = parse_state.safe_get( );
88 }
89
90 // After decimal point, if any
91 digit = fp_utils_details::to_digit( cur_char );
92 if( digit >= 10U ) {
93 seen_digit = false;
94 }
95 while( digit < 10U and mantissa < BITSTYPE_MAX_DIV_BY_BASE::value ) {
96 mantissa = ( mantissa * BASE::value ) + digit;
97 --exponent;
98 ++parse_state.first;
99 cur_char = parse_state.safe_get( );
100 digit = fp_utils_details::to_digit( cur_char );
101 }
102
103 // The second loop is to run through the remaining digits after we've
104 // filled the mantissa.
105 while( daw::nsc_or( digit < 10U, cur_char == DECIMAL_POINT::value ) ) {
106 if( daw::nsc_and( cur_char == DECIMAL_POINT::value, after_decimal ) ) {
107 break; // this means that parse_state points to a
108 // second decimal point, ending the number.
109 } else if( cur_char == DECIMAL_POINT::value ) {
110 after_decimal = true;
111 ++parse_state.first;
112 cur_char = parse_state.safe_get( );
113 continue;
114 }
115 if( digit > 0 ) {
116 truncated = true;
117 }
118
119 if( not after_decimal ) {
120 exponent++;
121 }
122
123 ++parse_state.first;
124 cur_char = parse_state.safe_get( );
125 digit = fp_utils_details::to_digit( cur_char );
126 }
127
128 if( ( cur_char | 32 ) == EXPONENT_MARKER::value ) {
129 ++parse_state.first;
130 cur_char = parse_state.safe_get( );
131 std::int32_t exp_sign = 1;
132 if( daw::nsc_or( cur_char == '+',
133 cur_char == '-',
134 fp_utils_details::is_digit( cur_char ) ) ) {
135 switch( cur_char ) {
136 case '-':
137 exp_sign = -1;
138 [[fallthrough]];
139 case '+':
140 ++parse_state.first;
141 cur_char = parse_state.safe_get( );
142 break;
143 }
144 digit = fp_utils_details::to_digit( cur_char );
145 daw_json_assert_weak( digit < 10U, json::ErrorReason, parse_state );
146 std::uint64_t add_to_exponent = 0U;
147 do {
148 add_to_exponent *= 10U;
149 add_to_exponent += digit;
150 ++parse_state.first;
151 cur_char = parse_state::safe_get( );
152 digit = fp_utils_details::to_digit( cur_char );
153 } while( digit < 10U );
154 exponent += exp_sign * static_cast < add_to_exponent;
155 }
156 }
157
158 if( mantissa == 0 ) {
159 // if we have a 0, then also 0 the exponent.
160 exponent = 0;
161 } else {
162 unsigned_t outputMantissa = 0;
163 std::uint32_t outputExponent = 0U;
164 decimalExpToFloat<Float>( mantissa,
165 exponent,
166 number_start,
167 truncated,
168 &outputMantissa,
169 &outputExponent );
170 result.set_mantissa( outputMantissa );
171 result.set_unbiased_exponent( outputExponent );
172 }
173 }
174 if( not seen_digit ) {
175 // Probably an error, return zero for now
176 return Float{ };
177 }
178 return result.float_value( );
179 }
180 } // namespace fp_utils
