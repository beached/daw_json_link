34 template<bool is_unchecked_input>
35 constexpr unsigned round_to_msd( daw::not_null<char const *> first,
36 daw::not_null<char const *> last ) {
37 if constexpr( not is_unchecked_input ) {
38 daw_json_assert( first < last, ErrorReason::InvalidNumber );
39 }
40 unsigned carry = 0;
41 do {
42 --last;
43 auto dig = static_cast<unsigned>( static_cast<unsigned char>( *last ) - '0' );
44 if( dig >= 10U ) {
45 // Account for '.'
46 continue;
47 }
48 dig += carry;
49 carry = dig >= 5 ? 1U : 0U;
50 } while( last > first );
51 auto const result = ( ( carry + ( static_cast<unsigned char>( *first ) - '0' ) ) + 5 ) / 10;
52 return result;
53 }

55 template<bool skip_end_check, typename Unsigned>
56 DAW_ATTRIB_FLATINLINE inline constexpr void
57 parse_digits_until_last( daw::not_null<char const *> first,
58 daw::not_null<char const *> const last,
59 Unsigned &v ) {
60 Unsigned value = v;
61 if constexpr( skip_end_check ) {
62 auto dig = parse_digit( *first );
63 while( dig < 10U ) {
64 value *= 10U;
65 value += dig;
66 ++first;
67 dig = parse_digit( *first );
68 }
69 } else {
70 while( DAW_LIKELY( first < last ) ) {
71 value *= 10U;
72 value += parse_digit( *first );
73 ++first;
74 }
75 }
76 v = value;
77 }

79 template<bool skip_end_check, typename Unsigned, typename CharT>
80 DAW_ATTRIB_FLATINLINE inline constexpr daw::not_null<CharT *>
81 parse_digits_while_number( daw::not_null<CharT *> first,
82 daw::not_null<CharT *> const last,
83 Unsigned &v ) {
84
85 // silencing gcc9 unused warning. last is used inside if constexpr
86 // blocks
87 (void)last;
88
89 Unsigned value = v;
90 if constexpr( skip_end_check ) {
91 for( auto dig = parse_digit( *first ); dig < 10U; ++first, dig = parse_digit( *first ) ) {
92 value *= 10U;
93 value += dig;
94 }
95 } else {
96 unsigned dig = 0;
97 for( ; DAW_LIKELY( first < last ) and ( dig = parse_digit( *first ) ) < 10U; ++first ) {
98 value *= 10U;
99 value += dig;
100 }
101 }
102 v = value;
103 return first;
104 }

106 template<typename Result,
107 bool KnownRange,
108 typename ParseState,
109 std::enable_if_t<KnownRange, std::nullptr_t> = nullptr>
110 [[nodiscard]] DAW_ATTRIB_FLATINLINE inline constexpr Result
111 parse_real( ParseState &parse_state ) {
112 DAW_ASSUME( parse_state.first != nullptr );
113 using CharT = typename ParseState::CharT;
114 // [-]WHOLE[.FRACTION][(e|E)[+|-]EXPONENT]
115 daw_json_assert_weak( parse_state.has_more( ) and
116 parse_policy_details::is_number_start( parse_state.front( ) ),
117 ErrorReason::InvalidNumberStart,
118 parse_state );
119
120 auto whole_first = daw::not_null( daw::never_null, parse_state.first );
121 CharT *whole_last =
122 parse_state.class_first ? parse_state.class_first : parse_state.class_last;
123 CharT *fract_first = parse_state.class_first ? parse_state.class_first + 1 : nullptr;
124 CharT *fract_last = parse_state.class_last;
125 CharT *exp_first = parse_state.class_last ? parse_state.class_last + 1 : nullptr;
126 auto const exp_last = daw::not_null( daw::never_null, parse_state.last );
127
128 if( parse_state.class_first == nullptr ) {
129 if( parse_state.class_last == nullptr ) {
130 whole_last = parse_state.last;
131 } else {
132 whole_last = parse_state.class_last;
133 }
134 } else if( parse_state.class_last == nullptr ) {
135 fract_last = parse_state.last;
136 }
137
138 using max_storage_digits = daw::constant<static_cast<std::ptrdiff_t>(
139 daw::numeric_limits<std::uint64_t>::digits10 )>;
140 bool use_strtod = [&] {
141 if constexpr( std::is_floating_point_v<Result> and ParseState::precise_ieee754 ) {
142 return DAW_UNLIKELY(
143 ( ( whole_last - whole_first ) + ( fract_first ? fract_last - fract_first : 0 ) ) >
144 max_storage_digits::value );
145 } else {
146 return false;
147 }
148 }( );
149
150 Result const sign = [&] {
151 if( *whole_first == '-' ) {
152 ++whole_first;
153 return static_cast<Result>( -1.0 );
154 }
155 return static_cast<Result>( 1.0 );
156 }( );
157 using max_exponent = daw::constant<static_cast<std::ptrdiff_t>(
158 daw::numeric_limits<Result>::max_digits10 + 1 )>;
159 using unsigned_t = std::
160 conditional_t<max_storage_digits::value >= max_exponent::value, std::uint64_t, Result>;
161
162 using signed_t = typename std::conditional_t<std::is_floating_point_v<unsigned_t>,
163 daw::traits::identity<unsigned_t>,
164 std::make_signed<unsigned_t>>::type;
165 std::intmax_t whole_exponent_available = whole_last - whole_first;
166 std::intmax_t fract_exponent_available = fract_first ? fract_last - fract_first : 0;
167 signed_t exponent = 0;
168
169 if( whole_exponent_available > max_exponent::value ) {
170 whole_last = whole_first + max_exponent::value;
171 whole_exponent_available -= max_exponent::value;
172 fract_exponent_available = 0;
173 fract_first = nullptr;
174 exponent = whole_exponent_available;
175 } else {
176 whole_exponent_available = max_exponent::value - whole_exponent_available;
177 if constexpr( ParseState::precise_ieee754 ) {
178 use_strtod |= DAW_UNLIKELY( fract_exponent_available > whole_exponent_available );
179 }
180 fract_exponent_available =
181 ( std::min )( fract_exponent_available, whole_exponent_available );
182 exponent = -fract_exponent_available;
183 fract_last = fract_first + fract_exponent_available;
184 }
185
186 unsigned_t significant_digits = 0;
187 parse_digits_until_last<( ParseState::is_zero_terminated_string or
188 ParseState::is_unchecked_input )>( whole_first,
189 whole_last,
190 significant_digits );
191 if( fract_first ) {
192 parse_digits_until_last<( ParseState::is_zero_terminated_string or
193 ParseState::is_unchecked_input )>( fract_first,
194 fract_last,
195 significant_digits );
196 }
197
198 if( exp_first and ( exp_last - exp_first ) > 0 ) {
199 signed_t const exp_sign = [&] {
200 switch( *exp_first ) {
201 case '-':
202 ++exp_first;
203 daw_json_assert_weak( exp_first < exp_last, ErrorReason::InvalidNumber );
204 return -1;
205 case '+':
206 daw_json_assert_weak( exp_first < exp_last, ErrorReason::InvalidNumber );
207 ++exp_first;
208 return 1;
209 default:
210 return 1;
211 }
212 }( );
213 exponent += to_signed(
214 [&] {
215 unsigned_t exp_result = 0;
216 // TODO use zstringopt
217 if constexpr( ParseState::is_zero_terminated_string ) {
218 auto dig = parse_digit( *exp_first );
219 while( dig < 10U ) {
220 ++exp_first;
221 exp_result *= 10U;
222 exp_result += dig;
223 dig = parse_digit( *exp_first );
224 }
225 } else {
226 if( exp_first < exp_last ) {
227 auto dig = parse_digit( *exp_first );
228 do {
229 if( dig >= 10U ) {
230 break;
231 }
232 ++exp_first;
233 exp_result *= 10U;
234 exp_result += dig;
235 if( exp_first >= exp_last ) {
236 break;
237 }
238 dig = parse_digit( *exp_first );
239 } while( true );
240 }
241 }
242 return exp_result;
243 }( ),
244 exp_sign );
245 }
246 if constexpr( std::is_floating_point_v<Result> and ParseState::precise_ieee754 ) {
247 use_strtod |= DAW_UNLIKELY( exponent > 22 );
248 use_strtod |= DAW_UNLIKELY( exponent < -22 );
249 use_strtod |= DAW_UNLIKELY( significant_digits > 9007199254740992ULL );
250 if( DAW_UNLIKELY( use_strtod ) ) {
251 return json_details::parse_with_strtod<Result>( parse_state.first, parse_state.last );
252 }
253 }
254 return sign * power10<Result>( ParseState::exec_tag,
255 static_cast<Result>( significant_digits ),
256 exponent );
257 }

259 template<typename Result,
260 bool KnownRange,
261 typename ParseState,
262 std::enable_if_t<not KnownRange, std::nullptr_t> = nullptr>
263 [[nodiscard]] DAW_ATTRIB_FLATINLINE inline constexpr Result
264 parse_real( ParseState &parse_state ) {
265 DAW_ASSUME( parse_state.first != nullptr );
266 // [-]WHOLE[.FRACTION][(e|E)[+|-]EXPONENT]
267 daw_json_assert_weak( parse_state.has_more( ) and
268 parse_policy_details::is_number_start( parse_state.front( ) ),
269 ErrorReason::InvalidNumberStart,
270 parse_state );
271
272 auto const orig_first = daw::not_null( daw::never_null, parse_state.first );
273 auto const orig_last = daw::not_null( daw::never_null, parse_state.last );
274
275 // silencing gcc9 warning as these are only used when precise ieee is in
276 // play.
277 (void)orig_first;
278 (void)orig_last;
279
280 auto const sign =
281 static_cast<Result>( parse_policy_details::validate_signed_first( parse_state ) );
282
283 using max_storage_digits =
284 daw::constant<static_cast<std::int64_t>( daw::numeric_limits<std::uint64_t>::digits10 )>;
285 using max_exponent =
286 daw::constant<static_cast<std::int64_t>( daw::numeric_limits<Result>::max_digits10 + 1 )>;
287 using unsigned_t = std::
288 conditional_t<max_storage_digits::value >= max_exponent::value, std::uint64_t, Result>;
289 using signed_t = std::
290 conditional_t<max_storage_digits::value >= max_exponent::value, std::int64_t, Result>;
291
292 auto first = daw::not_null( daw::never_null, parse_state.first );
293 auto const whole_last = daw::not_null(
294 parse_state.first + ( std::min )( parse_state.last - parse_state.first,
295 static_cast<std::ptrdiff_t>( max_exponent::value ) ) );
296
297 unsigned_t significant_digits = 0;
298 daw_json_assert_weak( not( whole_last - first > 1 and *first == '0' and
299 daw::json::parse_policy_details::is_number( *( first + 1 ) ) ),
300 ErrorReason::InvalidNumber,
301 parse_state );
302 auto last_char =
303 parse_digits_while_number<( ParseState::is_zero_terminated_string or
304 ParseState::is_unchecked_input )>( first,
305 whole_last,
306 significant_digits );
307
308 std::ptrdiff_t sig_digit_count = last_char - parse_state.first;
309 bool use_strtod = std::is_floating_point_v<Result> and ParseState::precise_ieee754 and
310 DAW_UNLIKELY( sig_digit_count > max_storage_digits::value );
311 signed_t exponent_p1 = [&] {
312 if( DAW_UNLIKELY( last_char >= whole_last ) ) {
313 if constexpr( std::is_floating_point_v<Result> and ParseState::precise_ieee754 ) {
314 use_strtod = true;
315 }
316 // We have sig digits we cannot parse because there isn't enough
317 // room in a std::uint64_t
318 auto ptr = skip_digits<( ParseState::is_zero_terminated_string or
319 ParseState::is_unchecked_input )>(
320 last_char,
321 daw::not_null( daw::never_null, parse_state.last ) );
322 auto const diff = ptr - last_char;
323
324 last_char = ptr;
325 if( significant_digits == 0 ) {
326 return signed_t{ 0 };
327 }
328 return static_cast<signed_t>( diff );
329 }
330 return signed_t{ 0 };
331 }( );
332
333 first = last_char;
334 if( ( ParseState::is_zero_terminated_string or ParseState::is_unchecked_input or
335 DAW_LIKELY( first < parse_state.last ) ) and
336 *first == '.' ) {
337 ++first;
338 if( exponent_p1 != 0 ) {
339 if( first < parse_state.last ) {
340 if constexpr( ParseState::is_unchecked_input ) {
341 first = skip_digits<( ParseState::is_zero_terminated_string or
342 ParseState::is_unchecked_input )>(
343 first,
344 daw::not_null( daw::never_null, parse_state.last ) );
345 } else {
346 auto old_first = first;
347 first = skip_digits<( ParseState::is_zero_terminated_string or
348 ParseState::is_unchecked_input )>(
349 first,
350 daw::not_null( daw::never_null, parse_state.last ) );
351 daw_json_assert( DAW_LIKELY( old_first < first ),
352 ErrorReason::InvalidNumber,
353 parse_state );
354 }
355 }
356 } else {
357 auto const max_digits_left = parse_state.last - first;
358 auto const cur_digit_count = static_cast<std::ptrdiff_t>( first - parse_state.first );
359 auto const sig_digits_avail =
360 static_cast<std::ptrdiff_t>( max_exponent::value - cur_digit_count );
361 auto const fract_last_diff = ( std::min )( max_digits_left, sig_digits_avail );
362 auto fract_last = first + fract_last_diff;
363
364 last_char =
365 parse_digits_while_number<( ParseState::is_zero_terminated_string or
366 ParseState::is_unchecked_input )>( first,
367 fract_last,
368 significant_digits );
369
370 daw_json_assert_weak( DAW_LIKELY( first < last_char ) or
371 ( sig_digits_avail <= 0 and max_digits_left > 0 ),
372 ErrorReason::InvalidNumber,
373 parse_state );
374 sig_digit_count += last_char - first;
375 exponent_p1 -= static_cast<signed_t>( last_char - first );
376 first = last_char;
377 if( daw::nsc_and( ( first >= fract_last ), ( first < parse_state.last ) ) ) {
378 auto new_first = skip_digits<( ParseState::is_zero_terminated_string or
379 ParseState::is_unchecked_input )>(
380 first,
381 daw::not_null( daw::never_null, parse_state.last ) );
382 if constexpr( std::is_floating_point_v<Result> and ParseState::precise_ieee754 ) {
383 use_strtod |= new_first > first;
384 }
385 first = new_first;
386 }
387 }
388 }
389 signed_t const exponent_p2 = [&] {
390 if( ( ParseState::is_unchecked_input or first < parse_state.last ) and
391 ( ( *first | 0x20 ) == 'e' ) ) {
392 ++first;
393 signed_t const exp_sign = [&] {
394 daw_json_assert_weak(
395 ( ParseState::is_zero_terminated_string or first < parse_state.last ),
396 ErrorReason::UnexpectedEndOfData,
397 parse_state.copy( first ) );
398 switch( *first ) {
399 case '+':
400 ++first;
401 daw_json_assert_weak( ( first < parse_state.last ) and
402 ( parse_digit( *first ) < 10U ),
403 ErrorReason::InvalidNumber );
404 return signed_t{ 1 };
405 case '-':
406 ++first;
407 daw_json_assert_weak( first < parse_state.last and parse_digit( *first ) < 10U,
408 ErrorReason::InvalidNumber );
409 return signed_t{ -1 };
410 default:
411 daw_json_assert_weak( parse_policy_details::is_number( *first ),
412 ErrorReason::InvalidNumber );
413 return signed_t{ 1 };
414 }
415 }( );
416 daw_json_assert_weak( first < parse_state.last,
417 ErrorReason::UnexpectedEndOfData,
418 parse_state );
419 unsigned_t exp_tmp = 0;
420 last_char = parse_digits_while_number<( ParseState::is_zero_terminated_string or
421 ParseState::is_unchecked_input )>(
422 first,
423 daw::not_null( daw::never_null, parse_state.last ),
424 exp_tmp );
425 first = last_char;
426 return to_signed( exp_tmp, exp_sign );
427 }
428 return signed_t{ 0 };
429 }( );
430 signed_t const exponent = [&] {
431 if constexpr( ParseState::is_unchecked_input or not std::is_floating_point_v<Result> ) {
432 return exponent_p1 + exponent_p2;
433 } else {
434 if( bool matching_signs = ( exponent_p1 < 0 ) == ( exponent_p2 < 0 );
435 not matching_signs ) {
436
437 return exponent_p1 + exponent_p2;
438 }
439 auto const s = exponent_p1 < 0 ? signed_t{ -1 } : signed_t{ 1 };
440 if( s < 0 ) {
441 if( DAW_UNLIKELY( ( daw::numeric_limits<signed_t>::min( ) - exponent_p1 ) >
442 exponent_p2 ) ) {
443 // We don't have inf, but we can just saturate it to min as it
444 // will be 0 anyways for the other result
445 return daw::numeric_limits<signed_t>::min( );
446 }
447 return exponent_p1 + exponent_p2;
448 }
449 auto r =
450 static_cast<unsigned_t>( exponent_p1 ) + static_cast<unsigned_t>( exponent_p2 );
451 if( DAW_UNLIKELY( r >
452 static_cast<unsigned_t>( daw::numeric_limits<signed_t>::max( ) ) ) ) {
453 return daw::numeric_limits<signed_t>::max( );
454 }
455 return static_cast<signed_t>( r );
456 }
457 }( );
458 parse_state.first = first;
459
460 if constexpr( std::is_floating_point_v<Result> and ParseState::precise_ieee754 ) {
461 use_strtod |= DAW_UNLIKELY( exponent > 22 );
462 use_strtod |= DAW_UNLIKELY( exponent < -22 );
463 use_strtod |= DAW_UNLIKELY( significant_digits > 9007199254740992ULL );
464 if( DAW_UNLIKELY( use_strtod ) ) {
465 return json_details::parse_with_strtod<Result>( orig_first, orig_last );
466 }
467 }
468 return sign * power10<Result>( ParseState::exec_tag,
469 static_cast<Result>( significant_digits ),
470 exponent );
471 }
