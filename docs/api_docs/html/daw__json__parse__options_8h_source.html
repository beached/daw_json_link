268 template<typename Policy>
269 inline constexpr json_options_t parse_policy_bits_start =
270 basic_policy_bits_start<Policy, parse_policy_list>;
271
272 template<typename Policy>
273 constexpr void set_bits_in( json_options_t &value, Policy e ) {
274 static_assert( is_option_flag<Policy>, "Only registered policy types are allowed" );
275 auto new_bits = static_cast<json_options_t>( e );
276 constexpr json_options_t mask = (1U << json_option_bits_width<Policy>)-1U;
277 new_bits &= mask;
278 new_bits <<= parse_policy_bits_start<Policy>;
279 value &= ~mask;
280 value |= new_bits;
281 }
282
283 template<typename Policy, typename... Policies>
284 constexpr json_options_t set_bits( json_options_t value, Policy pol, Policies... pols ) {
285 static_assert( ( is_option_flag<Policies> and ... ),
286 "Only registered policy types are allowed" );
287
288 auto new_bits = static_cast<json_options_t>( pol );
289 constexpr json_options_t mask = ( (1U << json_option_bits_width<Policy>)-1U );
290 new_bits &= mask;
291 new_bits <<= parse_policy_bits_start<Policy>;
292 value &= ~( mask << parse_policy_bits_start<Policy> );
293 value |= new_bits;
294 if constexpr( sizeof...( Policies ) > 0 ) {
295 if constexpr( sizeof...( pols ) > 0 ) {
296 return set_bits( value, pols... );
297 } else {
298 return value;
299 }
300 } else {
301 return value;
302 }
303 }
304
305 template<typename Policy>
306 constexpr json_options_t set_bits_for( Policy e ) {
307 static_assert( is_option_flag<Policy>, "Only registered policy types are allowed" );
308 auto new_bits = static_cast<json_options_t>( e );
309 new_bits <<= parse_policy_bits_start<Policy>;
310 return new_bits;
311 }
312
313 template<typename>
314 struct default_policy_flag_t;
315
316 template<typename... Policies>
317 struct default_policy_flag_t<pack_list<Policies...>> {
318 static constexpr json_options_t value =
319 ( set_bits_for<Policies>( default_json_option_value<Policies> ) | ... );
320 };
321
322 /***
323 * The defaults for all known policies encoded as a json_options_t
324 */
325 inline static constexpr json_options_t default_policy_flag =
326 default_policy_flag_t<parse_policy_list>::value;
327
328 template<typename Policy, typename Result = Policy>
329 constexpr Result get_bits_for( json_options_t value ) {
330 static_assert( is_option_flag<Policy>, "Only registered policy types are allowed" );
331 constexpr json_options_t mask =
332 ( 1U << (parse_policy_bits_start<Policy> + json_option_bits_width<Policy>)) - 1U;
333 value &= mask;
334 value >>= parse_policy_bits_start<Policy>;
335 return static_cast<Result>( Policy{ value } );
336 }

340 /***
341 * Create the parser options flag for BasicParsePolicy
342 * @tparam Policies Policy types that satisfy the `is_policy_flag` trait.
343 * @param policies A list of parser options to change from the defaults.
344 * @return A json_options_t that encodes the options for the parser
345 */
346 template<typename... Policies>
347 constexpr json_details::json_options_t parse_options( Policies... policies ) {
348 static_assert( ( json_details::is_option_flag<Policies> and ... ),
349 "Only registered policy types are allowed" );
350 auto result = json_details::default_policy_flag;
351 if constexpr( sizeof...( Policies ) > 0 ) {
352 // result |= ( json_details::set_bits_for( policies ) | ... );
353 ( json_details::set_nth_bits<
354 json_details::basic_option_bits_start<Policies, json_details::parse_policy_list>,
355 json_details::json_option_bits_width<Policies>>(
356 result,
357 static_cast<json_details::json_options_t>( policies ) ),
358 ... );
359 }
360 return result;
361 }
