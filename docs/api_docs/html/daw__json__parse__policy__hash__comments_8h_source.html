30 template<typename ParseState>
31 DAW_ATTRIB_FLATINLINE static constexpr void
32 skip_comments_unchecked( ParseState &parse_state ) {
33 while( parse_state.front( ) == '#' ) {
34 parse_state.remove_prefix( );
35 parse_state.template move_to_next_of_unchecked<'\n'>( );
36 parse_state.remove_prefix( );
37 }
38 }

40 template<typename ParseState>
41 DAW_ATTRIB_FLATINLINE static constexpr void skip_comments_checked( ParseState &parse_state ) {
42 while( parse_state.has_more( ) and parse_state.front( ) == '#' ) {
43 parse_state.remove_prefix( );
44 parse_state.template move_to_next_of_checked<'\n'>( );
45 if( parse_state.front( ) == '\n' ) {
46 parse_state.remove_prefix( );

85 DAW_ATTRIB_FLATINLINE static constexpr void move_to_next_of( ParseState &parse_state ) {
86 skip_comments( parse_state );
87
88 daw_json_assert_weak( parse_state.has_more( ),
89 ErrorReason::UnexpectedEndOfData,
90 parse_state );
91 while( not parse_policy_details::in<keys...>( parse_state.front( ) ) ) {
92 daw_json_assert_weak( parse_state.has_more( ),
93 ErrorReason::UnexpectedEndOfData,
94 parse_state );
95 parse_state.remove_prefix( );
96 skip_comments( parse_state );
97 }

104 template<char PrimLeft, char PrimRight, char SecLeft, char SecRight, typename ParseState>
105 DAW_ATTRIB_FLATINLINE static constexpr ParseState
106 skip_bracketed_item_checked( ParseState &parse_state ) {
107 // Not checking for Left as it is required to be skipped already
108 auto result = parse_state;
109 std::size_t cnt = 0;
110 std::uint32_t prime_bracket_count = 1;
111 std::uint32_t second_bracket_count = 0;
112 auto ptr_first = daw::not_null( daw::never_null, parse_state.first );
113 auto const ptr_last = daw::not_null( daw::never_null, parse_state.last );
114 if( DAW_UNLIKELY( ptr_first >= ptr_last ) ) {
115 return result;
116 }
117 if( *ptr_first == PrimLeft ) {
118 ++ptr_first;
119 }
120 while( DAW_LIKELY( ptr_first < ptr_last ) ) {
121 // TODO: use if/else if or put switch into IILE
122 switch( *ptr_first ) {
123 case '\\':
124 ++ptr_first;
125 break;
126 case '"':
127 ++ptr_first;
128 if constexpr( traits::not_same_v<typename ParseState::exec_tag_t,
129 constexpr_exec_tag> ) {
130 ptr_first =
131 json_details::mem_skip_until_end_of_string<ParseState::is_unchecked_input>(
132 ParseState::exec_tag,
133 ptr_first,
134 ptr_last );
135 } else {
136 while( DAW_LIKELY( ptr_first < ptr_last ) and *ptr_first != '"' ) {
137 if( *ptr_first == '\\' ) {
138 ++ptr_first;
139 if( ptr_first >= ptr_last ) {
140 break;
141 }
142 }
143 ++ptr_first;
144 }
145 }
146 daw_json_assert( ptr_first < ptr_last, ErrorReason::UnexpectedEndOfData, parse_state );
147 break;
148 case ',':
149 if( prime_bracket_count == 1 and second_bracket_count == 0 ) {
150 ++cnt;
151 }
152 break;
153 case PrimLeft:
154 ++prime_bracket_count;
155 break;
156 case PrimRight:
157 --prime_bracket_count;
158 if( prime_bracket_count == 0 ) {
159 daw_json_assert( second_bracket_count == 0,
160 ErrorReason::InvalidBracketing,
161 parse_state );
162 ++ptr_first;
163 // We include the close primary bracket in the range so that
164 // subsequent parsers have a terminator inside their range
165 result.last = ptr_first;
166 result.counter = cnt;
167 parse_state.first = ptr_first;
168 return result;
169 }
170 break;
171 case SecLeft:
172 ++second_bracket_count;
173 break;
174 case SecRight:
175 --second_bracket_count;
176 break;
177 case '#':
178 ++ptr_first;
179 while( ptr_first < ptr_last and *ptr_first != '\n' ) {
180 ++ptr_first;
181 }
182 if( ptr_first < ptr_last ) {
183 continue;
184 }
185 break;
186 }
187 ++ptr_first;
188 }
189 daw_json_assert( ( prime_bracket_count == 0 ) & ( second_bracket_count == 0 ),
190 ErrorReason::InvalidBracketing,
191 parse_state );
192 // We include the close primary bracket in the range so that subsequent
193 // parsers have a terminator inside their range
194 result.last = ptr_first;
195 result.counter = cnt;
196 parse_state.first = ptr_first;
197 return result;
198 }

200 template<char PrimLeft, char PrimRight, char SecLeft, char SecRight, typename ParseState>
201 DAW_ATTRIB_FLATINLINE static constexpr ParseState
202 skip_bracketed_item_unchecked( ParseState &parse_state ) {
203 // Not checking for Left as it is required to be skipped already
204 auto result = parse_state;
205 std::size_t cnt = 0;
206 std::uint32_t prime_bracket_count = 1;
207 std::uint32_t second_bracket_count = 0;
208 auto ptr_first = daw::not_null( daw::never_null, parse_state.first );
209 if( *ptr_first == PrimLeft ) {
210 ++ptr_first;
211 }
212 while( true ) {
213 switch( *ptr_first ) {
214 case '\\':
215 ++ptr_first;
216 break;
217 case '"':
218 ++ptr_first;
219 if constexpr( traits::not_same_v<typename ParseState::exec_tag_t,
220 constexpr_exec_tag> ) {
221 ptr_first =
222 json_details::mem_skip_until_end_of_string<ParseState::is_unchecked_input>(
223 ParseState::exec_tag,
224 ptr_first,
225 daw::not_null( daw::never_null, parse_state.last ) );
226 } else {
227 while( *ptr_first != '"' ) {
228 if( *ptr_first == '\\' ) {
229 ++ptr_first;
230 }
231 ++ptr_first;
232 }
233 }
234 break;
235 case ',':
236 if( prime_bracket_count == 1 and second_bracket_count == 0 ) {
237 ++cnt;
238 }
239 break;
240 case PrimLeft:
241 ++prime_bracket_count;
242 break;
243 case PrimRight:
244 --prime_bracket_count;
245 if( prime_bracket_count == 0 ) {
246 ++ptr_first;
247 // We include the close primary bracket in the range so that
248 // subsequent parsers have a terminator inside their range
249 result.last = ptr_first;
250 result.counter = cnt;
251 parse_state.first = ptr_first;
252 return result;
253 }
254 break;
255 case SecLeft:
256 ++second_bracket_count;
257 break;
258 case SecRight:
259 --second_bracket_count;
260 break;
261 case '#':
262 ++ptr_first;
263 while( *ptr_first != '\n' ) {
264 ++ptr_first;
265 }
266 break;
267 }
268 ++ptr_first;
269 }
270 DAW_UNREACHABLE( );
271 }
