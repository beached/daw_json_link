74 template<typename JsonElement, typename Container, typename OutputIterator>
75 [[maybe_unused]] constexpr OutputIterator to_json_array( Container const &c,
76 OutputIterator it ) {
77 static_assert( traits::is_container_like_v<daw::remove_cvref_t<Container>>,
78 "Supplied container must support begin( )/end( )" );
79 using iter_t = std::conditional_t<is_serialization_policy<OutputIterator>::value,
80 OutputIterator,
81 serialization_policy<OutputIterator>>;
82
83 auto out_it = iter_t( it );
84 if constexpr( std::is_pointer<OutputIterator>::value ) {
85 daw_json_assert( out_it, ErrorReason::NullOutputIterator );
86 }
87 *out_it++ = '[';
88 out_it.add_indent( );
89 // Not const & as some types(vector<bool>::const_reference are not ref
90 // types
91 auto first = std::begin( c );
92 auto last = std::end( c );
93 bool const has_elements = first != last;
94 while( first != last ) {
95 auto const &v = *first;
96 using v_type = DAW_TYPEOF( v );
97 using is_auto_detect = std::is_same<JsonElement, json_details::auto_detect_array_element>;
98 using JsonMember = std::conditional_t<is_auto_detect::value,
99 json_details::json_deduced_type<v_type>,
100 JsonElement>;
101
102 static_assert( not std::is_same_v<JsonMember, missing_json_data_contract_for<JsonElement>>,
103 "Unable to detect unnamed mapping" );
104 // static_assert( not std::is_same_v<JsonElement, JsonMember> );
105 out_it.next_member( );
106
107 out_it = json_details::member_to_string( template_arg<JsonMember>, out_it, v );
108 ++first;
109 if( first != last ) {
110 *out_it++ = ',';
111 }
112 }
113 // The last character will be a ',' prior to this
114 out_it.del_indent( );
115 if( has_elements ) {
116 out_it.output_newline( );
117 }
118 *out_it++ = ']';
119 return out_it.get( );
120 }
