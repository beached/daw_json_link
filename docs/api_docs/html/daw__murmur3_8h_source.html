25 [[nodiscard]] DAW_ATTRIB_FLATTEN inline constexpr UInt32 murmur3_32_scramble( UInt32 k ) {
26 using prime1 = daw::constant<0xcc9e'2d51_u32>;
27 using prime2 = daw::constant<0x1b87'3593_u32>;
28 k *= prime1::value;
29 k = rotate_left<15>( k );
30 k *= prime2::value;
31 return k;
32 }

45 template<bool expect_long_strings, typename StringView>
46 [[nodiscard]] DAW_ATTRIB_FLATTEN constexpr auto fnv1a_32( StringView key )
47 -> std::enable_if_t<daw::traits::is_string_view_like_v<StringView>, UInt32> {
48 std::size_t len = std::size( key );
49 auto ptr = daw::not_null( daw::never_null, std::data( std::as_const( key ) ) );
50 auto hash = 0x811c'9dc5_u32;
51 if constexpr( expect_long_strings ) {
52 while( len >= 8 ) {
53 hash = fnv1a_32_N<8>( ptr, hash );
54 len -= 8;
55 ptr += 8;
56 }
57 while( len >= 4 ) {
58 hash = fnv1a_32_N<4>( ptr, hash );
59 len -= 4;
60 ptr += 4;
61 }
62 }
63 for( std::size_t n = 0; n < len; ++n ) {
64 hash ^= static_cast<unsigned char>( ptr[n] );
65 hash *= 0x0100'0193_u32;
66 }
67 return hash;

71 [[nodiscard]] DAW_ATTRIB_INLINE constexpr auto name_hash( StringView key )
72 -> std::enable_if_t<daw::traits::is_string_view_like_v<StringView>, UInt32> {
73 if( auto const Sz = std::size( key ); DAW_LIKELY( Sz <= sizeof( UInt32 ) ) ) {
74 auto result = 0_u32;
75 auto ptr = daw::not_null( daw::never_null, std::data( std::as_const( key ) ) );
76 for( std::size_t n = 0; n < Sz; ++n ) {
77 result <<= 8U;

80 return result;

85 template<typename StringView>
86 [[nodiscard]] DAW_ATTRIB_FLATINLINE inline constexpr auto murmur3_32( StringView key,
87 std::uint32_t seed = 0 )
88 -> std::enable_if_t<daw::traits::is_string_view_like_v<StringView>, UInt32> {
89 UInt32 h = to_uint32( seed );
90 UInt32 k = 0_u32;
91 auto first = daw::not_null( daw::never_null, std::data( key ) );
92 auto const last = daw::not_null( daw::never_null, daw::data_end( key ) );
93 while( ( last - first ) >= 4 ) {
94 // Here is a source of differing results across endianness.
95 // A swap here has no effects on hash properties though.
96 k = daw::to_uint32_buffer( first );
97 first += 4;
98 h ^= murmur3_details::murmur3_32_scramble( k );
99 h = rotate_left<13>( h );
100 h = h * 5 + 0xe654'6b64_u32;
101 }
102
103 // Anything left over
104 k = 0_u32;
105 for( auto i = ( last - first ); i > 0; --i ) {
106 k <<= 8U;
107 k |= static_cast<unsigned char>( first[i - 1] );
108 }
109
110 h ^= murmur3_details::murmur3_32_scramble( k );
111
112 h ^= to_uint32( std::size( key ) );
113 h ^= h >> 16U;
114 h *= 0x85eb'ca6b_u32;
115 h ^= h >> 13U;
116 h *= 0xc2b2'ae35_u32;
117 h ^= h >> 16U;
118 return h;
119 }
