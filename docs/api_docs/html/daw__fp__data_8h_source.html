21 template<typename Float>
22 struct fp_data {
23 static_assert( std::is_floating_point_v<Float>, "Only floating point types are supported" );
24
25 // Reinterpreting bits as an integer value and interpreting the bits of
26 // an integer value as a floating point value is used in tests. So, a
27 // convenient type is provided for such reinterpretations.
28 using fp_prop = fp_properties<Float>;
29 using unsigned_t = typename fp_prop::unsigned_t;
30
31 unsigned_t bits{ };
32
33 constexpr void set_mantissa( unsigned_t mant_val ) noexcept {
34 mant_val &= ( fp_prop::mantissa_mask );
35 bits &= ~( fp_prop::mantissa_mask );
36 bits |= mant_val;
37 }
38
39 constexpr unsigned_t get_mantissa( ) const noexcept {
40 return bits & fp_prop::mantissa_mask;
41 }
42
43 constexpr void set_unbiased_exponent( unsigned_t expVal ) noexcept {
44 expVal = ( expVal << ( fp_prop::mantissa_width ) ) & fp_prop::exponent_mask;
45 bits &= ~( fp_prop::exponent_mask );
46 bits |= expVal;
47 }
48
49 constexpr uint16_t get_unbiased_exponent( ) const noexcept {
50 return uint16_t( ( bits & fp_prop::exponent_mask ) >> ( fp_prop::mantissa_width ) );
51 }
52
53 constexpr void set_sign( bool sign_val ) noexcept {
54 bits &= ~( fp_prop::sign_mask );
55 unsigned_t sign = unsigned_t( sign_val ) << ( fp_prop::bit_width - 1 );
56 bits |= sign;
57 }
58
59 constexpr bool get_sign( ) const noexcept {
60 return ( ( bits & fp_prop::sign_mask ) >> ( fp_prop::bit_width - 1 ) );
61 }
62
63 static_assert( sizeof( Float ) == sizeof( unsigned_t ),
64 "Data type and integral representation have different sizes." );
65
66 static constexpr int exponent_bias =
67 ( 1 << ( fp_prop::template exponent_width<Float>::value - 1 ) ) - 1;
68 static constexpr int max_exponent =
69 ( 1 << fp_prop::template exponent_width<Float>::value ) - 1;
70
71 static constexpr unsigned_t min_subnormal = unsigned_t{ 1U };
72 static constexpr unsigned_t max_subnormal =
73 ( unsigned_t{ 1U } << fp_prop::template mantissa_width<Float>::value ) - 1;
74 static constexpr unsigned_t min_normal =
75 ( unsigned_t{ 1U } << fp_prop::template mantissa_width<Float>::value );
76 static constexpr unsigned_t max_normal =
77 ( ( unsigned_t( max_exponent ) - 1 ) << fp_prop::template mantissa_width<Float>::value ) |
78 max_subnormal;
79
80 // Use SFINAE to prevent accidental type promotions/conversions
81 template<typename F, std::enable_if_t<std::is_same_v<Float, F>, std::nullptr_t> = nullptr>
82 explicit constexpr fp_data( F float_value ) noexcept
83 : bits( DAW_BIT_CAST( Float, float_value ) ) {}
84
85 template<typename U,
86 std::enable_if_t<std::is_same_v<unsigned_t, U>, std::nullptr_t> = nullptr>
87 explicit constexpr fp_data( U unsigned_value ) noexcept
88 : bits( unsigned_value ) {}
89
90 explicit constexpr fp_data( ) = default;
91
92 constexpr unsigned_t unsigned_value( ) const noexcept {
93 return bits;
94 }
95
96 constexpr unsigned_t float_value( ) const noexcept {
97 return DAW_BIT_CAST( Float, bits );
98 }
99
100 constexpr int get_exponent( ) const noexcept {
101 return int( get_unbiased_exponent( ) ) - exponent_bias;
102 }
103
104 constexpr bool is_zero( ) const noexcept {
105 return get_mantissa( ) == 0 and get_unbiased_exponent( ) == 0;
106 }
107
108 constexpr bool is_inf( ) const noexcept {
109 return get_mantissa( ) == 0 and get_unbiased_exponent( ) == max_exponent;
110 }
111
112 constexpr bool is_NaN( ) const noexcept {
113 return get_unbiased_exponent( ) == max_exponent and get_mantissa( ) != 0;
114 }
115
116 constexpr bool is_inf_or_NaN( ) const noexcept {
117 return get_unbiased_exponent( ) == max_exponent;
118 }
119
120 static constexpr fp_data zero( ) noexcept {
121 return fp_data( );
122 }
123
124 static constexpr fp_data neg_zero( ) noexcept {
125 return fp_data( unsigned_t{ 1U } << ( sizeof( unsigned_t ) * CHAR_BIT - 1U ) );
126 }
127
128 static constexpr fp_data inf( ) noexcept {
129 auto bits = fp_data{ };
130 bits.setUnbiased_exponent( max_exponent );
131 return bits;
132 }
133
134 static constexpr fp_data neg_inf( ) noexcept {
135 auto bits = inf( );
136 bits.set_sign( 1 );
137 return bits;
138 }
139
140 static constexpr Float build_NaN( unsigned_t v ) noexcept {
141 auto bits = inf( );
142 bits.set_mantissa( v );
143 return bits.float_value( );
144 }
145 };
