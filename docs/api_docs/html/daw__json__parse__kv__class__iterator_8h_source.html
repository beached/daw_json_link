32 template<typename ParseState>
33 struct json_parse_kv_class_iterator_base<ParseState, true> {
34 // We have to lie so that std::distance uses O(1) instead of O(N)
35 using iterator_category = std::random_access_iterator_tag;
36 using difference_type = std::ptrdiff_t;
37 ParseState *parse_state = nullptr;
38
39 constexpr difference_type operator-( json_parse_kv_class_iterator_base const &rhs ) const {
40 if( rhs.parse_state ) {
41 return static_cast<difference_type>( rhs.parse_state->counter );
42 }
43 return 0;
44 }

61 struct json_parse_kv_class_iterator :

64 using base = json_parse_kv_class_iterator_base<ParseState, can_random_v<IsKnown>>;
65 using iterator_category = typename base::iterator_category;
66 using element_t = typename JsonMember::json_element_t;
67 using member_container_type = typename JsonMember::base_type;

72 using iterator_range_t = ParseState;

78 inline constexpr json_parse_kv_class_iterator( ) = default;
79
80 inline constexpr explicit json_parse_kv_class_iterator( iterator_range_t &r )
81 : base{ &r } {
82 if( base::parse_state->front( ) == '}' ) {
83 // Cleanup at end of value
84 if( not IsKnown ) {
85 base::parse_state->remove_prefix( );
86 base::parse_state->trim_left_checked( );
87 // Ensure we are equal to default
88 }
89 base::parse_state = nullptr;
90 }
91 }
92
93 inline constexpr value_type operator*( ) {
94 daw_json_assert_weak( base::parse_state and base::parse_state->has_more( ),
95 ErrorReason::UnexpectedEndOfData,
96 *base::parse_state );
97 auto key = parse_value<key_t>( *base::parse_state, ParseTag<key_t::expected_type>{ } );
98 name::name_parser::trim_end_of_name( *base::parse_state );
99
100 return json_class_constructor<value_type, default_constructor<value_type>>(
101 DAW_MOVE( key ),
102 parse_value<value_t>( *base::parse_state, ParseTag<value_t::expected_type>{ } ) );
103 }
104
105 inline constexpr json_parse_kv_class_iterator &operator++( ) {
106 daw_json_assert_weak( base::parse_state,
107 ErrorReason::AttemptToAccessPastEndOfValue,
108 *base::parse_state );
109 base::parse_state->move_next_member_or_end( );
110 daw_json_assert_weak( base::parse_state->has_more( ),
111 ErrorReason::UnexpectedEndOfData,
112 *base::parse_state );
113 if( base::parse_state->front( ) == '}' ) {
114#ifndef NDEBUG
115 if constexpr( IsKnown ) {
116 if( base::parse_state ) {
117 daw_json_assert( base::parse_state->counter > 0,
118 ErrorReason::AttemptToAccessPastEndOfValue,
119 *base::parse_state );
120 base::parse_state->counter--;
121 }
122 }
123#endif
124 if constexpr( not IsKnown ) {
125 // Cleanup at end of value
126 base::parse_state->remove_prefix( );
127 base::parse_state->trim_left_checked( );
128 // Ensure we are equal to default
129 }
130 base::parse_state = nullptr;
131 }
132#ifndef NDEBUG
133 if constexpr( IsKnown ) {
134 if( base::parse_state ) {
135 daw_json_assert( base::parse_state->counter > 0,
136 ErrorReason::AttemptToAccessPastEndOfValue,
137 *base::parse_state );
138 base::parse_state->counter--;
139 }
140 }
141#endif
142 return *this;
143 }
144
145 friend inline constexpr bool operator==( json_parse_kv_class_iterator const &lhs,
146 json_parse_kv_class_iterator const &rhs ) {
147 // using identity as equality
148 return lhs.parse_state == rhs.base::parse_state;
149 }
150
151 friend inline constexpr bool operator!=( json_parse_kv_class_iterator const &lhs,
152 json_parse_kv_class_iterator const &rhs ) {
153 return not( lhs == rhs );
154 }
