101 /***
102 * This is the default ToJsonConverter for json_custom. By default is will
103 * return the stringified version of the value if, to_string( T ) exists.
104 * Otherwise it will fallback to an std::ostream converter for T if it
105 * exists.
106 * @tparam T type of value to convert to a string
107 */
108 template<typename T>
109 struct default_to_json_converter_t {
110 template<typename U>
111 [[nodiscard]] static inline auto use_stream( U const &v ) {
112 std::stringstream ss{ };
113 ss << v;
114 return DAW_MOVE( ss ).str( );
115 }
116
117 template<typename U>
118 [[nodiscard]] inline constexpr auto operator( )( U const &value ) const {
119 if constexpr( json_details::is_string_view_like_v<U> ) {
120 return std::string_view( std::data( value ), std::size( value ) );
121 } else if constexpr( json_details::to_strings::has_to_string_v<U> ) {
122 using json_details::to_strings::to_string;
123 return to_string( value );
124 } else if constexpr( json_details::has_ostream_op_v<U> ) {
125 return use_stream( value );
126 } else if constexpr( std::is_convertible_v<U, std::string_view> ) {
127 return static_cast<std::string_view>( value );
128 } else if constexpr( std::is_convertible_v<U, std::string> ) {
129 return static_cast<std::string>( value );
130 } else if constexpr( daw::is_arithmetic_v<U> ) {
131 return to_string( value );
132 } else if constexpr( std::is_enum_v<U> ) {
133 return to_string( static_cast<std::underlying_type_t<U>>( value ) );
134 } else if constexpr( json_details::is_dereferenceable_v<U> ) {
135 static_assert( json_details::has_op_bool_v<U>,
136 "default_to_converter cannot work with type" );
137 using dereferenced_type = json_details::dereferenced_t<U>;
138 if constexpr( json_details::is_string_view_like_v<dereferenced_type> ) {
139 if( value ) {
140 auto const &v = *value;
141 return std::string_view( std::data( v ), std::size( v ) );
142 }
143 return std::string_view( "null", 4 );
144 } else if constexpr( json_details::to_strings::has_to_string_v<dereferenced_type> ) {
145 if( value ) {
146 using json_details::to_strings::to_string;
147 return to_string( *value );
148 } else {
149 using result_t = DAW_TYPEOF( to_string( *value ) );
150 return result_t{ "null" };
151 }
152 } else if constexpr( std::is_convertible_v<dereferenced_type, std::string_view> ) {
153 if( value ) {
154 return static_cast<std::string_view>( value );
155 }
156 return std::string_view{ "null" };
157 } else if constexpr( std::is_convertible_v<dereferenced_type, std::string> ) {
158 if( value ) {
159 return static_cast<std::string>( value );
160 }
161 return std::string( "null" );
162 } else {
163 if( value ) {
164 return use_stream( *value );
165 } else {
166 return std::string( "null" );
167 }
168 }
169 }
170 }
171 };

184 template<typename T>
185 struct default_from_json_converter_t {
186 [[nodiscard]] inline constexpr decltype( auto ) operator( )( std::string_view sv ) const {
187 if constexpr( std::disjunction<std::is_same<T, std::string_view>,
188 std::is_same<T, std::optional<std::string_view>>>::value ) {
189 return sv;
190 } else if constexpr( json_details::has_from_string_v<T> ) {
191 return from_string( daw::tag<T>, sv );
192 } else if constexpr( std::is_convertible_v<std::string_view, T> ) {
193 return static_cast<T>( sv );
194 } else if constexpr( std::is_convertible_v<std::string, T> ) {
195 return static_cast<T>( static_cast<std::string>( sv ) );
196 } else {
197 static_assert( json_details::has_istream_op_v<T>,
198 "Unsupported type in default to converter. Must "
199 "supply a custom one" );
200 static_assert( std::is_default_constructible_v<T>,
201 "Unsupported type in default to converter. Must "
202 "supply a custom one" );
203 return from_json_conv_details::use_stream<T>( sv );
204 }
205 }

329 template<typename OutputIterator>
330 constexpr void utf32_to_utf8( std::uint32_t cp, OutputIterator &it ) {
331 if( cp <= 0x7FU ) {
332 *it++ = static_cast<char>( cp );
333 return;
334 }
335 if( cp <= 0x7FFU ) {
336 *it++ = static_cast<char>( ( cp >> 6U ) | 0b11000000U );
337 *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
338 return;
339 }
340 if( cp <= 0xFFFFU ) {
341 *it++ = static_cast<char>( ( cp >> 12U ) | 0b11100000U );
342 *it++ = static_cast<char>( ( ( cp >> 6U ) & 0b00111111U ) | 0b10000000U );
343 *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
344 return;
345 }
346 if( cp <= 0x10FFFFU ) {
347 *it++ = static_cast<char>( ( cp >> 18U ) | 0b11110000U );
348 *it++ = static_cast<char>( ( ( cp >> 12U ) & 0b00111111U ) | 0b10000000U );
349 *it++ = static_cast<char>( ( ( cp >> 6U ) & 0b00111111U ) | 0b10000000U );
350 *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
351 return;
352 }
353 daw_json_error( ErrorReason::InvalidUTFCodepoint );
354 }

358 template<bool do_escape = false,
359 EightBitModes EightBitMode = EightBitModes::AllowFull,
360 typename OutputIterator,
361 typename Container,
362 std::enable_if_t<traits::is_container_like_v<daw::remove_cvref_t<Container>>,
363 std::nullptr_t> = nullptr>
364 [[nodiscard]] constexpr OutputIterator copy_to_iterator( OutputIterator it,
365 Container const &container ) {
366 constexpr bool restrict_high = EightBitMode != EightBitModes::AllowFull or
367 ( OutputIterator::restricted_string_output ==
368 RestrictedStringOutput::OnlyAllow7bitsStrings );
369 if constexpr( do_escape ) {
370 using iter = DAW_TYPEOF( std::begin( container ) );
371 using it_t = utf8::unchecked::iterator<iter>;
372 auto first = it_t( std::begin( container ) );
373 auto const last = it_t( std::end( container ) );
374 while( first != last ) {
375 auto const cp = *first++;
376 switch( cp ) {
377 case '"':
378 *it++ = '\\';
379 *it++ = '"';
380 break;
381 case '\\':
382 *it++ = '\\';
383 *it++ = '\\';
384 break;
385 case '\b':
386 *it++ = '\\';
387 *it++ = 'b';
388 break;
389 case '\f':
390 *it++ = '\\';
391 *it++ = 'f';
392 break;
393 case '\n':
394 *it++ = '\\';
395 *it++ = 'n';
396 break;
397 case '\r':
398 *it++ = '\\';
399 *it++ = 'r';
400 break;
401 case '\t':
402 *it++ = '\\';
403 *it++ = 't';
404 break;
405 default:
406 if( cp < 0x20U ) {
407 it = json_details::output_hex( static_cast<std::uint16_t>( cp ), it );
408 break;
409 }
410 if constexpr( restrict_high ) {
411 if( cp >= 0x7FU and cp <= 0xFFFFU ) {
412 it = json_details::output_hex( static_cast<std::uint16_t>( cp ), it );
413 break;
414 }
415 if( cp > 0xFFFFU ) {
416 it =
417 json_details::output_hex( static_cast<std::uint16_t>( 0xD7C0U + ( cp >> 10U ) ),
418 it );
419 it = json_details::output_hex(
420 static_cast<std::uint16_t>( 0xDC00U + ( cp & 0x3FFU ) ),
421 it );
422 break;
423 }
424 }
425 json_details::utf32_to_utf8( cp, it );
426 break;
427 }
428 }
429 } else {
430 for( auto c : container ) {
431 if constexpr( restrict_high ) {
432 daw_json_assert( ( static_cast<unsigned char>( c ) >= 0x20U and
433 static_cast<unsigned char>( c ) <= 0x7FU ),
434 ErrorReason::InvalidStringHighASCII );
435 }
436 *it++ = c;
437 }
438 }
439 return it;
440 }

620 template<typename JsonMember, typename OutputIterator, typename parse_to_t>
621 [[nodiscard]] constexpr OutputIterator to_daw_json_string( ParseTag<JsonParseTypes::Real>,
622 OutputIterator it,
623 parse_to_t const &value ) {
624
625 static_assert( std::is_convertible<parse_to_t, json_result<JsonMember>>::value,
626 "value must be convertible to specified type in class contract" );
627
628 if constexpr( std::is_floating_point<json_result<JsonMember>>::value ) {
629 if( daw::cxmath::is_nan( value ) ) {
630 if constexpr( JsonMember::literal_as_string == LiteralAsStringOpt::Never or
631 JsonMember::allow_number_errors == JsonNumberErrors::None or
632 JsonMember::allow_number_errors == JsonNumberErrors::AllowInf ) {
633 daw_json_error( ErrorReason::NumberIsNaN );
634 } else {
635 *it++ = '"';
636 it = utils::copy_to_iterator( it, "NaN" );
637 *it++ = '"';
638 return it;
639 }
640 } else if( daw::cxmath::is_inf( value ) ) {
641 if constexpr( JsonMember::literal_as_string == LiteralAsStringOpt::Never or
642 JsonMember::allow_number_errors == JsonNumberErrors::None or
643 JsonMember::allow_number_errors == JsonNumberErrors::AllowNaN ) {
644 daw_json_error( ErrorReason::NumberIsInf );
645 } else {
646 *it++ = '"';
647 if( value < 0 ) {

704 "value must be convertible to specified type in class contract" );

707 using to_strings::to_string;

712 }

745 *ptr++ = static_cast<char>( '0' + static_cast<char>( v ) );

748 *it++ = *ptr;

1488
1489 auto const br = [&] {
1490 if constexpr( std::is_same_v<Real, float> ) {
1491 return daw::jkj::dragonbox::ieee754_bits( value );
1492 } else {
1493 return daw::jkj::dragonbox::ieee754_bits( static_cast<double>( value ) );
1494 }
1495 }( );

1497 if( br.is_negative( ) ) {
1498 *out_it++ = '-';
1499 }

1502 }
1503 if constexpr( true and fp_output_fmt == FPOutputFormat::Scientific ) {
1504 return daw::jkj::dragonbox::to_chars_detail::to_chars( dec, out_it, digits );
1505 } else /*if constexpr( fp_output_fmt != FPOutputFormat::Decimal )*/ {
1506 if( ( whole_dig < -4 ) | ( whole_dig > 6 ) ) {
1507 return daw::jkj::dragonbox::to_chars_detail::to_chars( dec, out_it, digits );
1508 }
1509 }
1510 if( br.is_negative( ) ) {
1511 *out_it++ = '-';
1512 }

1514 *out_it++ = '0';

1516 }
1517 if( dec.exponent < 0 ) {
1518
1519 if( whole_dig < 0 ) {
1520 *out_it++ = '0';
1521 *out_it++ = '.';
1522 do {
1523 *out_it++ = '0';
1524 ++whole_dig;
1525 } while( whole_dig < 0 );

1527 return out_it;
1528 }

1530 auto const p1pow = daw::cxmath::pow10( static_cast<std::size_t>( -dec.exponent ) );
