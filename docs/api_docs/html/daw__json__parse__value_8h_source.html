47 /***
48 * Depending on the type of literal, it may always be quoted, sometimes,
49 * or never. This method handles the always and sometimes cases.
50 * In checked input, ensures State has more data.
51 * @tparam literal_as_string Is the literal being parsed enclosed in
52 * quotes.
53 * @tparam ParseState ParseState idiom
54 * @param parse_state Current parsing state
55 */
56 template<LiteralAsStringOpt literal_as_string, typename ParseState>
57 DAW_ATTRIB_INLINE constexpr void
58 skip_quote_when_literal_as_string( ParseState &parse_state ) {
59 if constexpr( literal_as_string == LiteralAsStringOpt::Always ) {
60 daw_json_assert_weak( parse_state.is_quotes_checked( ),
61 ErrorReason::InvalidNumberUnexpectedQuoting,
62 parse_state );
63 parse_state.remove_prefix( );
64 } else if constexpr( literal_as_string == LiteralAsStringOpt::Maybe ) {
65 daw_json_assert_weak( parse_state.has_more( ),
66 ErrorReason::UnexpectedEndOfData,
67 parse_state );
68 if( parse_state.front( ) == '"' ) {
69 parse_state.remove_prefix( );
70 }
71 }
72 }

74 template<typename JsonMember, bool KnownBounds, typename ParseState>
75 [[nodiscard, maybe_unused]] DAW_ATTRIB_INLINE constexpr json_result<JsonMember>
76 parse_value( ParseState &parse_state, ParseTag<JsonParseTypes::Real> ) {
77 using constructor_t = typename JsonMember::constructor_t;
78 using element_t = typename JsonMember::base_type;
79
80 if constexpr( JsonMember::literal_as_string != LiteralAsStringOpt::Never ) {
81 skip_quote_when_literal_as_string<JsonMember::literal_as_string>( parse_state );
82 if constexpr( JsonMember::allow_number_errors == JsonNumberErrors::AllowInf or
83 JsonMember::allow_number_errors == JsonNumberErrors::AllowNanInf ) {
84 element_t sign = element_t( 1.0 );
85 if( parse_state.front( ) == '-' ) {
86 sign = element_t( -1.0 );
87 parse_state.first++;
88 }
89 // Looking for Inf as that will match Infinity too.
90 if( parse_state.size( ) >= 4 and parse_state.starts_with( "Inf" ) ) {
91
92 parse_state.first += 3;
93 if( parse_state.front( ) == '"' ) {
94 parse_state.first++;
95 } else if( parse_state.size( ) >= 6 and parse_state.starts_with( R"(inity")" ) ) {
96 parse_state.first += 6;
97 } else {
98 daw_json_error( ErrorReason::InvalidString, parse_state );
99 }
100 daw_json_assert_weak( not parse_state.has_more( ) or
101 parse_policy_details::at_end_of_item( parse_state.front( ) ),
102 ErrorReason::InvalidEndOfValue,
103 parse_state );
104 return daw::cxmath::copy_sign( daw::numeric_limits<element_t>::infinity( ), sign );
105 } else if( sign < element_t( 0 ) ) {
106 parse_state.first--;
107 }
108 }
109 if constexpr( JsonMember::allow_number_errors == JsonNumberErrors::AllowNaN or
110 JsonMember::allow_number_errors == JsonNumberErrors::AllowNanInf ) {
111 if( parse_state.starts_with( "NaN" ) ) {
112 parse_state.template move_to_next_of<'"'>( );
113 parse_state.remove_prefix( );
114 daw_json_assert_weak( not parse_state.has_more( ) or
115 parse_policy_details::at_end_of_item( parse_state.front( ) ),
116 ErrorReason::InvalidEndOfValue,
117 parse_state );
118 return daw::numeric_limits<element_t>::quiet_NaN( );
119 }
120 }
121 }
122 if constexpr( KnownBounds ) {
123 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
124 parse_state,
125 parse_real<element_t, true>( parse_state ) );
126 } else {
127 daw_json_assert_weak( parse_state.has_more( ) and
128 parse_policy_details::is_number_start( parse_state.front( ) ),
129 ErrorReason::InvalidNumberStart,
130 parse_state );
131
132 auto result = construct_value( template_args<json_result<JsonMember>, constructor_t>,
133 parse_state,
134 parse_real<element_t, false>( parse_state ) );
135 if constexpr( JsonMember::literal_as_string != LiteralAsStringOpt::Never ) {
136 skip_quote_when_literal_as_string<JsonMember::literal_as_string>( parse_state );
137 }
138
139 daw_json_assert_weak( not parse_state.has_more( ) or
140 parse_policy_details::at_end_of_item( parse_state.front( ) ),
141 ErrorReason::InvalidEndOfValue,
142 parse_state );
143 return result;
144 }
145 }

149 parse_value( ParseState &parse_state, ParseTag<JsonParseTypes::Signed> ) {
150 using constructor_t = typename JsonMember::constructor_t;
151 using element_t = typename JsonMember::base_type;
152 using int_type = typename std::conditional_t<std::is_enum_v<element_t>,
153 std::underlying_type<element_t>,
154 daw::traits::identity<element_t>>::type;
155
156 static_assert( daw::is_signed<int_type>::value, "Expected signed type" );
157 if constexpr( KnownBounds ) {
158 daw_json_assert_weak( parse_policy_details::is_number_start( parse_state.front( ) ),
159 ErrorReason::InvalidNumberStart,
160 parse_state );
161 } else {
162 if constexpr( JsonMember::literal_as_string != LiteralAsStringOpt::Never ) {
163 skip_quote_when_literal_as_string<JsonMember::literal_as_string>( parse_state );

166 ErrorReason::UnexpectedEndOfData,
167 parse_state );
168 }
169 }
170 auto const sign =
171 static_cast<int_type>( parse_policy_details::validate_signed_first( parse_state ) );
172 using uint_type = typename std::conditional_t<daw::is_system_integral_v<int_type>,

174 daw::traits::identity<int_type>>::type;

176 unsigned_parser<uint_type, JsonMember::range_check, KnownBounds>( ParseState::exec_tag,
177 parse_state ),
178 sign );
179
180 if constexpr( KnownBounds ) {
181 return construct_value( template_args<json_result<JsonMember>, constructor_t>,

183 static_cast<element_t>( parsed_val ) );
184 } else {
185 auto result = construct_value( template_args<json_result<JsonMember>, constructor_t>,
186 parse_state,
187 static_cast<element_t>( parsed_val ) );
188 if constexpr( JsonMember::literal_as_string != LiteralAsStringOpt::Never ) {
189 skip_quote_when_literal_as_string<JsonMember::literal_as_string>( parse_state );
190 }
191 parse_state.trim_left( );
192 daw_json_assert_weak( not parse_state.has_more( ) or
193 parse_policy_details::at_end_of_item( parse_state.front( ) ),
194 ErrorReason::InvalidEndOfValue,
195 parse_state );
196 return result;

258 [[nodiscard, maybe_unused]] DAW_ATTRIB_INLINE constexpr json_result<JsonMember>
259 parse_value( ParseState &parse_state, ParseTag<JsonParseTypes::Null> ) {
260
261 using constructor_t = typename JsonMember::constructor_t;
262 if constexpr( KnownBounds ) {
263 // skip_value will leave a null parse_state
264 if( parse_state.is_null( ) ) {
265 if constexpr( JsonMember::nullable == JsonNullable::Nullable ) {
266 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
267 parse_state );
268 } else {
269 daw_json_error( missing_member( "non-nullable member" ) );
270 }
271 }
272 return parse_value<JsonMember, true>( parse_state,
273 ParseTag<JsonMember::base_expected_type>{ } );
274 } else if constexpr( ParseState::is_unchecked_input ) {
275 if( not parse_state.has_more( ) ) {
276 if constexpr( JsonMember::nullable == JsonNullable::Nullable ) {
277 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
278 parse_state );
279 } else {
280 daw_json_error( missing_member( std::string_view( std::data( JsonMember::name ),
281 std::size( JsonMember::name ) ) ) );
282 }
283 } else if( parse_state.front( ) == 'n' ) {
284 parse_state.remove_prefix( 4 );
285 parse_state.trim_left_unchecked( );
286 parse_state.remove_prefix( );
287 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
288 parse_state );
289 }
290 return parse_value<JsonMember>( parse_state,
291 ParseTag<JsonMember::base_expected_type>{ } );
292 } else {
293 if( parse_state.is_null( ) ) {
294 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
295 parse_state );
296 }
297 if( parse_state.is_null( ) or parse_state.starts_with( "null" ) ) {
298 parse_state.remove_prefix( 4 );
299 daw_json_assert_weak( not parse_state.has_more( ) or
300 parse_policy_details::at_end_of_item( parse_state.front( ) ),
301 ErrorReason::InvalidLiteral,
302 parse_state );
303 parse_state.trim_left_checked( );
304 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
305 parse_state );
306 }
307 return parse_value<JsonMember>( parse_state,
308 ParseTag<JsonMember::base_expected_type>{ } );
309 }

312 template<typename JsonMember, bool KnownBounds, typename ParseState>
313 [[nodiscard, maybe_unused]] DAW_ATTRIB_INLINE constexpr json_result<JsonMember>
314 parse_value( ParseState &parse_state, ParseTag<JsonParseTypes::Bool> ) {
315 using constructor_t = typename JsonMember::constructor_t;
316
317 if constexpr( KnownBounds ) {
318 // We have already checked if it is a true/false
319 if constexpr( ParseState::is_unchecked_input ) {
320 return static_cast<bool>( parse_state.counter );
321 } else {
322 switch( parse_state.front( ) ) {
323 case 't':
324 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
325 parse_state,
326 true );
327 case 'f':
328 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
329 parse_state,
330 false );
331 }
332 daw_json_error( ErrorReason::InvalidLiteral, parse_state );
333 }
334 } else {
335 // Beginning quotes
336 if constexpr( JsonMember::literal_as_string != LiteralAsStringOpt::Never ) {
337 skip_quote_when_literal_as_string<JsonMember::literal_as_string>( parse_state );
338 }
339 bool result = false;
340 if constexpr( ParseState::is_unchecked_input ) {
341 if( parse_state.front( ) == 't' ) /* true */ {
342 result = true;
343 parse_state.remove_prefix( 4 );
344 } else /* false */ {
345 parse_state.remove_prefix( 5 );
346 }
347 } else {
348 if( parse_state.starts_with( "true" ) ) {
349 parse_state.remove_prefix( 4 );
350 result = true;
351 } else if( parse_state.starts_with( "false" ) ) {
352 parse_state.remove_prefix( 5 );
353 } else {
354 daw_json_error( ErrorReason::InvalidLiteral, parse_state );
355 }
356 }
357 // Trailing quotes
358 if constexpr( JsonMember::literal_as_string != LiteralAsStringOpt::Never ) {
359 skip_quote_when_literal_as_string<JsonMember::literal_as_string>( parse_state );
360 }
361 parse_state.trim_left( );
362 daw_json_assert_weak( not parse_state.has_more( ) or
363 parse_policy_details::at_end_of_item( parse_state.front( ) ),
364 ErrorReason::InvalidEndOfValue,
365 parse_state );
366 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
367 parse_state,
368 result );
369 }
370 }

372 template<typename JsonMember, bool KnownBounds, typename ParseState>
373 [[nodiscard, maybe_unused]] DAW_ATTRIB_INLINE constexpr json_result<JsonMember>
374 parse_value( ParseState &parse_state, ParseTag<JsonParseTypes::StringRaw> ) {
375
376 using constructor_t = typename JsonMember::constructor_t;
377 if constexpr( KnownBounds ) {
378 if constexpr( JsonMember::empty_is_null == EmptyStringNull::yes ) {
379 if( parse_state.empty( ) ) {
380 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
381 parse_state );
382 }
383 }
384 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
385 parse_state,
386 std::data( parse_state ),
387 std::size( parse_state ) );
388 } else {
389 if constexpr( JsonMember::allow_escape_character == AllowEscapeCharacter::Allow ) {
390 auto const str = skip_string( parse_state );
391 if constexpr( JsonMember::empty_is_null == EmptyStringNull::yes ) {
392 if( str.empty( ) ) {
393 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
394 parse_state );
395 }
396 }
397 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
398 parse_state,
399 std::data( str ),
400 std::size( str ) );
401 } else {
402 DAW_ASSUME( parse_state.first != nullptr );
403 parse_state.remove_prefix( );
404
405 auto const first = daw::not_null( daw::never_null, parse_state.first );
406 parse_state.template move_to_next_of<'"'>( );
407 auto const last = daw::not_null( daw::never_null, parse_state.first );
408 parse_state.remove_prefix( );
409 if constexpr( JsonMember::empty_is_null == EmptyStringNull::yes ) {
410 if( first == last ) {
411 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
412 parse_state );
413 }
414 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
415 parse_state,
416 first,
417 static_cast<std::size_t>( last - first ) );
418 } else {
419 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
420 parse_state,
421 first,
422 static_cast<std::size_t>( last - first ) );
423 }
424 }
425 }

457 static_assert( has_json_member_parse_to_v<JsonMember> );

462 std::bool_constant<ParseState::eight_bit_mode == GlobalEightBitModes::AllowFull and

477 } else {

479 using AllowHighEightbits =
480 std::bool_constant<ParseState::eight_bit_mode == GlobalEightBitModes::AllowFull and

483 // There are escapes in the string
484 return parse_string_known_stdstring<AllowHighEightbits::value, JsonMember, true>(
485 parse_state2 );
486 }

514
515 auto const str = [&] {
516 if constexpr( JsonMember::custom_json_type == JsonCustomTypes::String ) {
517 if constexpr( KnownBounds ) {
518 return parse_state;
519 } else {
520 return skip_string( parse_state );

533 }
534 return result;

537 auto result = skip_string( parse_state );
538 result.first--;
539 return result;
540 }
541 return skip_value( parse_state );
542 }
543 }
544 }( );
545 daw_json_assert_weak( str.has_more( ) and not( str.front( ) == '[' or str.front( ) == '{' ),
546 ErrorReason::InvalidStartOfValue,
547 str );
548 using constructor_t = typename JsonMember::from_converter_t;
549 return construct_value( template_args<json_result<JsonMember>, constructor_t>,
550 parse_state,
551 std::string_view( std::data( str ), std::size( str ) ) );

576 [[nodiscard, maybe_unused]] DAW_ATTRIB_FLATTEN constexpr json_result<JsonMember>

580 daw_json_assert_weak( parse_state.has_more( ),
581 ErrorReason::UnexpectedEndOfData,
582 parse_state );
583
584 if constexpr( use_direct_construction_v<ParseState, JsonMember> ) {
585 // This relies on non-trivial dtor's being allowed. So C++20
586 // constexpr or not in a constant expression. It does allow for
587 // construction of classes without move/copy special members

594 (void)run_after_parse;
595 return json_data_contract_trait_t<element_t>::parse_to_class(
596 parse_state,
597 template_arg<JsonMember> );
598 }
599 } else {
600 if constexpr( KnownBounds ) {
601 return json_data_contract_trait_t<element_t>::parse_to_class(
