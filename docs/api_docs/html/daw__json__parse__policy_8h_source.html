306 template<std::size_t N>
307 inline constexpr bool starts_with( char const ( &rhs )[N] ) const {
308 if( size( ) < ( N - 1 ) ) {
309 return false;
310 }
311 bool result = true;
312 for( std::size_t n = 0; n < ( N - 1 ); ++n ) {
313 result = daw::nsc_and( result, ( first[n] == rhs[n] ) );
314 }
315 return result;
316 }

389 inline constexpr void move_to_end_of_literal( ) {
390 auto f = daw::not_null( daw::never_null, first );
391 auto const l = daw::not_null( daw::never_null, last );
392 if constexpr( is_unchecked_input ) {
393 while( daw::nsc_and( static_cast<unsigned char>( *f ) > 0x20U,
394 not CommentPolicy::is_literal_end( *f ) ) ) {
395 ++f;
396 }
397 } else {
398 while( ( f < l ) and daw::nsc_and( static_cast<unsigned char>( *f ) > 0x20U,
399 not CommentPolicy::is_literal_end( *f ) ) ) {
400 ++f;
401 }
402 }
403 first = f;
404 }
