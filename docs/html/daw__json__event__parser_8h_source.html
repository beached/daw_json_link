337  template<typename ParsePolicy = NoCommentSkippingPolicyChecked,
338  typename StackContainerPolicy = DefaultJsonEventParserStackPolicy<
339  JsonEventParserStackValue<ParsePolicy>>,
340  typename Handler>
341  inline constexpr void
342  json_event_parser( daw::json::basic_json_value<ParsePolicy> jvalue,
343  Handler &&handler ) {
344  using iterator = daw::json::basic_json_value_iterator<ParsePolicy>;
345  using json_value_t = daw::json::basic_json_pair<ParsePolicy>;
346  using stack_value_t = JsonEventParserStackValue<ParsePolicy>;
347 
348  auto parent_stack = StackContainerPolicy( );
349 
350  auto const move_to_last = [&]( ) {
351  parent_stack.back( ).value.first = parent_stack.back( ).value.second;
352  };
353 
354  auto const process_value = [&]( json_value_t p ) {
355  {
356  auto result = json_details::handle_on_value( handler, p );
357  switch( result.value ) {
358  case json_parse_handler_result::Complete:
359  parent_stack.clear( );
360  return;
361  case json_parse_handler_result::SkipClassArray:
362  move_to_last( );
363  return;
364  case json_parse_handler_result::Continue:
365  break;
366  }
367  }
368 
369  auto &jv = p.value;
370  switch( jv.type( ) ) {
371  case daw::json::JsonBaseParseTypes::Array: {
372  auto result = json_details::handle_on_array_start( handler, jv );
373  switch( result.value ) {
374  case json_parse_handler_result::Complete:
375  parent_stack.clear( );
376  return;
377  case json_parse_handler_result::SkipClassArray:
378  move_to_last( );
379  return;
380  case json_parse_handler_result::Continue:
381  break;
382  }
383  parent_stack.push_back(
384  { StackRangeType::Array,
385  std::pair<iterator, iterator>( jv.begin( ), jv.end( ) ) } );
386  } break;
387  case daw::json::JsonBaseParseTypes::Class: {
388  auto result = json_details::handle_on_class_start( handler, jv );
389  switch( result.value ) {
390  case json_parse_handler_result::Complete:
391  parent_stack.clear( );
392  return;
393  case json_parse_handler_result::SkipClassArray:
394  move_to_last( );
395  return;
396  case json_parse_handler_result::Continue:
397  break;
398  }
399  parent_stack.push_back(
400  { StackRangeType::Class,
401  std::pair<iterator, iterator>( jv.begin( ), jv.end( ) ) } );
402  } break;
403  case daw::json::JsonBaseParseTypes::Number: {
404  auto result = json_details::handle_on_number( handler, jv );
405  switch( result.value ) {
406  case json_parse_handler_result::Complete:
407  parent_stack.clear( );
408  return;
409  case json_parse_handler_result::SkipClassArray:
410  move_to_last( );
411  return;
412  case json_parse_handler_result::Continue:
413  break;
414  }
415  } break;
416  case daw::json::JsonBaseParseTypes::Bool: {
417  auto result = json_details::handle_on_bool( handler, jv );
418  switch( result.value ) {
419  case json_parse_handler_result::Complete:
420  parent_stack.clear( );
421  return;
422  case json_parse_handler_result::SkipClassArray:
423  move_to_last( );
424  return;
425  case json_parse_handler_result::Continue:
426  break;
427  }
428  } break;
429  case daw::json::JsonBaseParseTypes::String: {
430  auto result = json_details::handle_on_string( handler, jv );
431  switch( result.value ) {
432  case json_parse_handler_result::Complete:
433  parent_stack.clear( );
434  return;
435  case json_parse_handler_result::SkipClassArray:
436  move_to_last( );
437  return;
438  case json_parse_handler_result::Continue:
439  break;
440  }
441  } break;
442  case daw::json::JsonBaseParseTypes::Null: {
443  auto result = json_details::handle_on_null( handler, jv );
444  switch( result.value ) {
445  case json_parse_handler_result::Complete:
446  parent_stack.clear( );
447  return;
448  case json_parse_handler_result::SkipClassArray:
449  move_to_last( );
450  return;
451  case json_parse_handler_result::Continue:
452  break;
453  }
454  } break;
455  case daw::json::JsonBaseParseTypes::None:
456  default: {
457  auto result = json_details::handle_on_error( handler, jv );
458  switch( result.value ) {
459  case json_parse_handler_result::Complete:
460  parent_stack.clear( );
461  return;
462  case json_parse_handler_result::SkipClassArray:
463  move_to_last( );
464  return;
465  case json_parse_handler_result::Continue:
466  break;
467  }
468  } break;
469  }
470  };
471 
472  auto const process_range = [&]( stack_value_t v ) {
473  if( v.value.first != v.value.second ) {
474  auto jv = *v.value.first;
475  v.value.first++;
476  parent_stack.push_back( std::move( v ) );
477  process_value( std::move( jv ) );
478  } else {
479  switch( v.type ) {
480  case StackRangeType::Class: {
481  auto result = json_details::handle_on_class_end( handler );
482  switch( result.value ) {
483  case json_parse_handler_result::Complete:
484  parent_stack.clear( );
485  return;
486  case json_parse_handler_result::SkipClassArray:
487  case json_parse_handler_result::Continue:
488  break;
489  }
490  } break;
491  case StackRangeType::Array: {
492  auto result = json_details::handle_on_array_end( handler );
493  switch( result.value ) {
494  case json_parse_handler_result::Complete:
495  parent_stack.clear( );
496  return;
497  case json_parse_handler_result::SkipClassArray:
498  case json_parse_handler_result::Continue:
499  break;
500  }
501  } break;
502  }
503  }
504  };
505 
506  process_value( { std::nullopt, std::move( jvalue ) } );
507 
508  while( not parent_stack.empty( ) ) {
509  auto v = std::move( parent_stack.back( ) );
510  parent_stack.pop_back( );
511  process_range( v );
512  }
513  }
