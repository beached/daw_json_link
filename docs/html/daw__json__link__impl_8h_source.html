164  template<std::size_t start_pos>
165  [[nodiscard]] constexpr std::size_t
166  find_name( daw::string_view key ) const {
167  UInt32 const hash = name_hash( key );
168 #if defined( _MSC_VER ) and not defined( __clang__ )
169  // MSVC has a bug where the list initialization isn't sequenced in order
170  // of appearance.
171  for( std::size_t n = 0; n < N; ++n ) {
172 #else
173  for( std::size_t n = start_pos; n < N; ++n ) {
174 #endif
175  if( names[n].hash_value == hash ) {
176  if constexpr( has_collisons ) {
177  if( DAW_JSON_UNLIKELY( key != *names[n].name ) ) {
178  continue;
179  }
180  }
181  return n;
182  }
183  }
184  return N;
185  }

189  constexpr bool do_hashes_collide( ) noexcept {
190  std::array<UInt32, sizeof...( MemberNames )> hashes{
191  name_hash( MemberNames::name )... };

193  daw::sort( hashes.data( ), hashes.data( ) + hashes.size( ) );

196  []( UInt32 l, UInt32 r ) { return l == r; } ) != hashes.end( );

207  /***
208  * Get the position from already seen JSON members or move the parser
209  * forward until we reach the end of the class or the member.
210  * @tparam pos JsonMember's position in locations
211  * @tparam JsonMember current member in json_class
212  * @tparam N Number of members in json_class
213  * @tparam Range see IteratorRange
214  * @param locations members location and names
215  * @param rng Current JSON data
216  * @return IteratorRange with begin( ) being start of value
217  */
218  template<std::size_t pos, typename JsonMember, std::size_t N, typename Range,
219  bool B>
220  [[nodiscard]] inline constexpr Range
221  find_class_member( locations_info_t<N, Range, B> &locations, Range &rng ) {
222 
223  daw_json_assert_weak(
224  ( is_json_nullable_v<JsonMember> or not locations[pos].missing( ) or
225  not rng.is_closing_brace_checked( ) ),
226  "Unexpected end of class. Non-nullable members still not found", rng );
227 
228  rng.trim_left_unchecked( );
229  // TODO: should we check for end
230  while( locations[pos].missing( ) & ( rng.front( ) != '}' ) ) {
231  daw_json_assert_weak( rng.has_more( ), "Unexpected end of stream", rng );
232  // TODO: fully unescape name
233  auto const name = parse_name( rng );
234  auto const name_pos = locations.template find_name<pos>( name );
235  if( name_pos >= locations.size( ) ) {
236  // This is not a member we are concerned with
237  (void)skip_value( rng );
238  rng.clean_tail( );
239  continue;
240  }
241  if( name_pos == pos ) {
242  locations[pos].location = rng;
243  break;
244  } else {
245  // We are out of order, store position for later
246  // OLDTODO: use type knowledge to speed up skip
247  // OLDTODO: on skipped classes see if way to store
248  // member positions so that we don't have to
249  // reparse them after
250  // RESULT: storing preparsed is slower, don't try 3 times
251  // it also limits the type of things we can parse potentially
252  // Using locations to switch on BaseType is slower too
253  locations[name_pos].location = skip_value( rng );
254 
255  rng.clean_tail( );
256  }
257  }
258  return locations[pos].location;
259  }

287  /***
288  * Parse a class member in an order json class(class as array)
289  * @tparam JsonMember type description of member to parse
290  * @tparam Range see IteratorRange
291  * @param member_position current position in array
292  * @param rng JSON data
293  * @return A reified value of type JsonMember::parse_to_t
294  */
295  template<typename JsonMember, typename Range>
296  [[nodiscard]] DAW_ATTRIBUTE_FLATTEN inline constexpr auto
297  parse_ordered_class_member( std::size_t &member_position, Range &rng ) {
298 
299  /***
300  * Some members specify their index so there may be gaps between member
301  * data elements in the array.
302  */
303  if constexpr( is_an_ordered_member_v<JsonMember> ) {
304  pocm_details::maybe_skip_members( rng, member_position,
305  JsonMember::member_index );
306  } else {
307  rng.clean_tail( );
308  }
309  using json_member_type = ordered_member_subtype_t<JsonMember>;
310 
311  // this is an out value, get position ready
312  ++member_position;
313  if( DAW_JSON_UNLIKELY( rng.front( ) == ']' ) ) {
314  if constexpr( is_json_nullable_v<ordered_member_subtype_t<JsonMember>> ) {
315  using constructor_t = typename json_member_type::constructor_t;
316  return constructor_t{ }( );
317  } else if constexpr( is_json_nullable_v<json_member_type> ) {
318  daw_json_error( missing_member( "ordered_class_member" ) );
319  }
320  }
321  return parse_value<json_member_type>(
322  ParseTag<json_member_type::expected_type>{ }, rng );
323  }

325  /***
326  * Parse a member from a json_class
327  * @tparam member_position position in json_class member list
328  * @tparam JsonMember type description of member to parse
329  * @tparam N Number of members in json_class
330  * @tparam Range see IteratorRange
331  * @param locations location info for members
332  * @param rng JSON data
333  * @return parsed value from JSON data
334  */
335  template<std::size_t member_position, typename JsonMember, std::size_t N,
336  typename Range, bool B>
337  [[nodiscard]] DAW_ATTRIBUTE_FLATTEN inline constexpr json_result<JsonMember>
338  parse_class_member( locations_info_t<N, Range, B> &locations, Range &rng ) {
339  rng.clean_tail( );
340  static_assert( not is_no_name<JsonMember>,
341  "Array processing should never call parse_class_member" );
342 
343  daw_json_assert_weak( rng.is_at_next_class_member( ),
344  "Expected end of class or start of member", rng );
345  auto loc = find_class_member<member_position, JsonMember>( locations, rng );
346 
347  // If the member was found loc will have it's position
348  if( loc.first == rng.first ) {
349  return parse_value<JsonMember>( ParseTag<JsonMember::expected_type>{ },
350  rng );
351  }
352  if( not loc.is_null( ) ) {
353  return parse_value<JsonMember, true>(
354  ParseTag<JsonMember::expected_type>{ }, loc );
355  }
356  // We cannot find the member, check if the member is nullable
357  if constexpr( is_json_nullable_v<JsonMember> ) {
358  return parse_value<JsonMember, true>(
359  ParseTag<JsonMember::expected_type>{ }, loc );
360  } else {
361  daw_json_error( missing_member( std::string_view(
362  JsonMember::name.data( ), JsonMember::name.size( ) ) ) );
363  }
364  }

381  /***
382  * Parse to the user supplied class. The parser will run left->right if it
383  * can when the JSON document's order matches that of the order of the
384  * supplied classes ctor. If there is an order mismatch, store the
385  * start/finish of JSON members we are interested in and return that to the
386  * members parser when needed.
387  */
388  template<typename JsonClass, typename... JsonMembers, std::size_t... Is,
389  typename Range>
390  [[nodiscard]] inline constexpr JsonClass
391  parse_json_class( Range &rng, std::index_sequence<Is...> ) {
392  static_assert( has_json_data_contract_trait_v<JsonClass>,
393  "Unexpected type" );
394  rng.trim_left( );
395  daw_json_assert_weak( rng.is_opening_brace_checked( ),
396  "Expected start of class", rng );
397  rng.set_class_position( );
398  rng.remove_prefix( );
399  rng.move_to_next_class_member( );
400 
401  if constexpr( sizeof...( JsonMembers ) == 0 ) {
402  // Clang-CL with MSVC has issues if we don't do empties this way
403  class_cleanup_now( rng );
404  return daw::construct_a<JsonClass>( );
405  } else {
406 
407 #if not defined( _MSC_VER ) or defined( __clang__ )
408  // Prior to C++20, this will guarantee the data structure is initialized
409  // at compile time. In the future, constinit should be fine.
410  constexpr auto known_locations_v =
411  make_locations_info<Range, JsonMembers...>( );
412 
413  auto known_locations = known_locations_v;
414 #else
415  // MSVC is doing the murmur3 hash at compile time incorrectly
416  // this puts it at runtime.
417  auto known_locations = make_locations_info<Range, JsonMembers...>( );
418 
419 #endif
420  if constexpr( is_guaranteed_rvo_v<Range> ) {
421  struct cleanup_t {
422  Range *rng_ptr;
423  CPP20CONSTEXPR inline ~cleanup_t( ) {
424  class_cleanup_now( *rng_ptr );
425  }
426  } const run_after_parse{ &rng };
427  /*
428  * Rather than call directly use apply/tuple to evaluate left->right
429  */
430  if constexpr( daw::json::force_aggregate_constrution<
431  JsonClass>::value ) {
432  return JsonClass{
433  parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
434  known_locations, rng )... };
435  } else {
436  using tp_t = decltype( std::forward_as_tuple(
437  parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
438  known_locations, rng )... ) );
439  return std::apply(
440  daw::construct_a<JsonClass>,
441  tp_t{ parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
442  known_locations, rng )... } );
443  }
444  } else {
445  using tp_t = decltype( std::forward_as_tuple(
446  parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
447  known_locations, rng )... ) );
448  JsonClass result = std::apply(
449  daw::construct_a<JsonClass>,
450  tp_t{ parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
451  known_locations, rng )... } );
452  class_cleanup_now( rng );
453  return result;
454  }
455  }

458  /***
459  * Parse to a class where the members are constructed from the values of a
460  * JSON array. Often this is used for geometric types like Point
461  */
462  template<typename JsonClass, typename... JsonMembers, typename Range>
463  [[nodiscard]] inline constexpr JsonClass
464  parse_ordered_json_class( Range &rng ) {
465  static_assert( has_json_data_contract_trait_v<JsonClass>,

468  can_construct_a_v<JsonClass, typename JsonMembers::parse_to_t...>,

480  parse_ordered_class_member<JsonMembers>( current_idx, rng ) )...>;

483  struct cleanup_t {
