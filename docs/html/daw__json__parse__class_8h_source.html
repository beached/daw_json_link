34  * Maybe skip json members

36  * @param parse_state JSON data
37  * @param current_position current member index
38  * @param desired_position desired member index

40  template<bool Nullable, typename ParseState>
41  constexpr void maybe_skip_members( ParseState &parse_state,
42  std::size_t &current_position,
43  std::size_t desired_position ) {
44 
45  parse_state.clean_tail( );
46  daw_json_assert_weak( parse_state.has_more( ),
47  ErrorReason::UnexpectedEndOfData, parse_state );
48  daw_json_assert_weak( current_position <= desired_position,
49  ErrorReason::OutOfOrderOrderedMembers,
50  parse_state );
51  constexpr bool skip_check_end =
52  ParseState::is_unchecked_input and Nullable;
53  while( ( current_position < desired_position ) &
54  ( skip_check_end or parse_state.front( ) != ']' ) ) {
55  (void)skip_value( parse_state );
56  parse_state.clean_tail( );
57  ++current_position;
58  daw_json_assert_weak( parse_state.has_more( ),
59  ErrorReason::UnexpectedEndOfData,
60  parse_state );
61  }

65  /***
66  * Parse a class member in an order json class(class as array)
67  * @tparam JsonMember type description of member to parse
68  * @tparam ParseState see IteratorRange
69  * @param member_index current position in array
70  * @param parse_state JSON data
71  * @return A reified value of type JsonMember::parse_to_t
72  */
73  template<typename JsonMember, typename ParseState>
74  [[nodiscard]] constexpr json_result<JsonMember>
75  parse_ordered_class_member( std::size_t &member_index,
76  ParseState &parse_state ) {
77 
78  using json_member_t = ordered_member_subtype_t<JsonMember>;
79  /***
80  * Some members specify their index so there may be gaps between member
81  * data elements in the array.
82  */
83  if constexpr( is_an_ordered_member_v<JsonMember> ) {
84  pocm_details::maybe_skip_members<is_json_nullable_v<json_member_t>>(
85  parse_state, member_index, JsonMember::member_index );
86  } else {
87  parse_state.clean_tail( );
88  }
89 
90  // this is an out value, get position ready
91  ++member_index;
92  if constexpr( ParseState::is_unchecked_input ) {
93  if constexpr( is_json_nullable_v<json_member_t> ) {
94  if( parse_state.front( ) == ']' ) {
95  using constructor_t = typename json_member_t::constructor_t;
96  return constructor_t{ }( );
97  }
98  }
99  } else {
100  if( DAW_JSON_UNLIKELY( parse_state.front( ) == ']' ) ) {
101  if constexpr( is_json_nullable_v<json_member_t> ) {
102  using constructor_t = typename json_member_t::constructor_t;
103  return constructor_t{ }( );
104  } else {
105  daw_json_error( missing_member( "ordered_class_member" ),
106  parse_state );
107  }
108  }
109  }
110  return parse_value<json_member_t>(
111  ParseTag<json_member_t::expected_type>{ }, parse_state );
112  }

114  /***
115  * Parse a member from a json_class
116  * @tparam member_position position in json_class member list
117  * @tparam JsonMember type description of member to parse
118  * @tparam N Number of members in json_class
119  * @tparam ParseState see IteratorRange
120  * @param locations location info for members
121  * @param parse_state JSON data
122  * @return parsed value from JSON data
123  */
124  template<std::size_t member_position, typename JsonMember, std::size_t N,
125  typename ParseState, bool B>
126  [[nodiscard]] constexpr json_result<JsonMember>
127  parse_class_member( locations_info_t<N, B> &locations,
128  ParseState &parse_state ) {
129  parse_state.clean_tail( );
130  static_assert(
131  not is_no_name<JsonMember>,
132  "Array processing should never call parse_class_member" );
133 
134  daw_json_assert_weak( parse_state.is_at_next_class_member( ),
135  ErrorReason::MissingMemberNameOrEndOfClass,
136  parse_state );
137  ParseState loc = [&] {
138  if constexpr( ParseState::has_allocator ) {
139  return find_class_member<member_position>(
140  locations, parse_state, is_json_nullable_v<JsonMember>,
141  JsonMember::name )
142  .with_allocator( parse_state.get_allocator( ) );
143  } else {
144  return find_class_member<member_position>(
145  locations, parse_state, is_json_nullable_v<JsonMember>,
146  JsonMember::name );
147  }
148  }( );
149 
150  // If the member was found loc will have it's position
151  if( loc.first == parse_state.first ) {
152  return parse_value<JsonMember>(
153  ParseTag<JsonMember::expected_type>{ }, parse_state );
154  }
155  // We cannot find the member, check if the member is nullable
156  if constexpr( is_json_nullable_v<JsonMember> ) {
157  if( loc.is_null( ) ) {
158  return parse_value<JsonMember, true>(
159  ParseTag<JsonMember::expected_type>{ }, loc );
160  }
161  } else {
162  daw_json_assert_weak(
163  not loc.is_null( ),
164  missing_member( std::string_view( std::data( JsonMember::name ),
165  std::size( JsonMember::name ) ) ),
166  parse_state );
167  }
168  return parse_value<JsonMember, true>(
169  ParseTag<JsonMember::expected_type>{ }, loc );
170  }

186  /***
187  * Parse to the user supplied class. The parser will run left->right if
188  * it can when the JSON document's order matches that of the order of the
189  * supplied classes ctor. If there is an order mismatch, store the
190  * start/finish of JSON members we are interested in and return that to
191  * the members parser when needed.
192  */
193  template<typename JsonClass, typename... JsonMembers, std::size_t... Is,
194  typename ParseState>
195  [[nodiscard]] static constexpr json_result<JsonClass>
196  parse_json_class( ParseState &parse_state, std::index_sequence<Is...> ) {
197  static_assert( is_a_json_type_v<JsonClass> );
198  using T = typename JsonClass::base_type;
199  using Constructor = typename JsonClass::constructor_t;
200  static_assert( has_json_data_contract_trait_v<T>, "Unexpected type" );
201 
202  parse_state.trim_left( );
203  // TODO, use member name
204  daw_json_assert_weak( parse_state.is_opening_brace_checked( ),
205  ErrorReason::InvalidClassStart, parse_state );
206  parse_state.set_class_position( );
207  parse_state.remove_prefix( );
208  parse_state.trim_left( );
209 
210  if constexpr( sizeof...( JsonMembers ) == 0 ) {
211  // Clang-CL with MSVC has issues if we don't do empties this way
212  class_cleanup_now( parse_state );
213  return construct_value<T>( Constructor{ }, parse_state );
214  } else {
215 
216 #if not defined( _MSC_VER ) or defined( __clang__ )
217  // Prior to C++20, this will guarantee the data structure is
218  // initialized at compile time. In the future, constinit should be
219  // fine.
220  constexpr auto known_locations_v =
221  make_locations_info<ParseState, JsonMembers...>( );
222 
223  auto known_locations = known_locations_v;
224 #else
225  // MSVC is doing the murmur3 hash at compile time incorrectly
226  // this puts it at runtime.
227  auto known_locations =
228  make_locations_info<ParseState, JsonMembers...>( );
229 
230 #endif
231  if constexpr( is_guaranteed_rvo_v<ParseState> ) {
232  struct cleanup_t {
233  ParseState *parse_state_ptr;
234  CPP20CONSTEXPR
235  inline ~cleanup_t( ) noexcept( false ) {
236 #ifdef HAS_CPP20CONSTEXPR
237  if( not std::is_constant_evaluated( ) ) {
238 #endif
239  if( std::uncaught_exceptions( ) == 0 ) {
240  class_cleanup_now( *parse_state_ptr );
241  }
242 #ifdef HAS_CPP20CONSTEXPR
243  } else {
244  class_cleanup_now( *parse_state_ptr );
245  }
246 #endif
247  }
248  } const run_after_parse{ &parse_state };
249  (void)run_after_parse;
250  /*
251  * Rather than call directly use apply/tuple to evaluate left->right
252  */
253  if constexpr( force_aggregate_construction_v<T> ) {
254  return T{
255  parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
256  known_locations, parse_state )... };
257  } else {
258  using tp_t = decltype( std::forward_as_tuple(
259  parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
260  known_locations, parse_state )... ) );
261 
262  return construct_value_tp<T>(
263  Constructor{ }, parse_state,
264  tp_t{
265  parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
266  known_locations, parse_state )... } );
267  }
268  } else {
269  using tp_t = decltype( std::forward_as_tuple(
270  parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
271  known_locations, parse_state )... ) );
272  auto result = construct_value_tp<T>(
273  Constructor{ }, parse_state,
274  tp_t{
275  parse_class_member<Is, traits::nth_type<Is, JsonMembers...>>(
276  known_locations, parse_state )... } );
277  class_cleanup_now( parse_state );
278  return result;
279  }

283  /***
284  * Parse to a class where the members are constructed from the values of a
285  * JSON array. Often this is used for geometric types like Point
286  */
287  template<typename JsonClass, typename... JsonMembers, typename ParseState>
288  [[nodiscard]] static constexpr json_result<JsonClass>
289  parse_ordered_json_class( ParseState &parse_state ) {
290  static_assert( is_a_json_type_v<JsonClass> );
291  using T = typename JsonClass::base_type;
292  using Constructor = typename JsonClass::constructor_t;
293  static_assert( has_json_data_contract_trait_v<T>, "Unexpected type" );
294  static_assert(
295  std::is_invocable_v<Constructor, typename JsonMembers::parse_to_t...>,
296  "Supplied types cannot be used for construction of this type" );
297 
298  parse_state.trim_left( ); // Move to array start '['
299  daw_json_assert_weak( parse_state.is_opening_bracket_checked( ),
300  ErrorReason::InvalidArrayStart, parse_state );
301  parse_state.set_class_position( );
302  parse_state.remove_prefix( );
303  parse_state.trim_left( );
304 
305  size_t current_idx = 0;
306 
307  using tp_t = decltype(
308  std::forward_as_tuple( parse_ordered_class_member<JsonMembers>(
309  current_idx, parse_state )... ) );
310 
311  if constexpr( is_guaranteed_rvo_v<ParseState> ) {
312  struct cleanup_t {
313  ParseState *ptr;
314  CPP20CONSTEXPR inline ~cleanup_t( ) noexcept( false ) {
315 #ifdef HAS_CPP20CONSTEXPR
316  if( not std::is_constant_evaluated( ) ) {
317 #endif
318  if( std::uncaught_exceptions( ) == 0 ) {
319  (void)ptr->skip_array( );
320  }
321 #ifdef HAS_CPP20CONSTEXPR
322  } else {
323  (void)ptr->skip_array( );
324  }
325 #endif
326  }
327  } const run_after_parse{ &parse_state };
328  (void)run_after_parse;
329  return construct_value_tp<T>(
330  Constructor{ }, parse_state,
331  tp_t{ parse_ordered_class_member<JsonMembers>( current_idx,
332  parse_state )... } );
333  } else {
334  auto result = construct_value_tp<T>(
335  Constructor{ }, parse_state,
336  tp_t{ parse_ordered_class_member<JsonMembers>( current_idx,
337  parse_state )... } );
