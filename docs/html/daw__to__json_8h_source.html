41  template<typename JsonElement, typename Container, typename OutputIterator>
42  [[maybe_unused]] constexpr OutputIterator
43  to_json_array( Container const &c, OutputIterator out_it ) {
44  static_assert( traits::is_container_like_v<daw::remove_cvref_t<Container>>,
45  "Supplied container must support begin( )/end( )" );
46 
47  if constexpr( std::is_pointer_v<OutputIterator> ) {
48  daw_json_assert( out_it, ErrorReason::NullOutputIterator );
49  }
50  *out_it++ = '[';
51  bool is_first = true;
52  for( auto const &v : c ) {
53  using v_type = daw::remove_cvref_t<decltype( v )>;
54  constexpr bool is_auto_detect_v =
55  std::is_same_v<JsonElement, json_details::auto_detect_array_element>;
56  using JsonMember =
57  std::conditional_t<is_auto_detect_v,
58  json_details::unnamed_default_type_mapping<v_type>,
59  JsonElement>;
60 
61  static_assert(
62  traits::not_same_v<
63  JsonMember, daw::json::missing_json_data_contract_for<JsonElement>>,
64  "Unable to detect unnamed mapping" );
65  static_assert( traits::not_same_v<JsonElement, JsonMember> );
66  if( is_first ) {
67  is_first = false;
68  } else {
69  *out_it++ = ',';
70  }
71  out_it = json_details::member_to_string<JsonMember>( out_it, v );
72  }
73  // The last character will be a ',' prior to this
74  *out_it++ = ']';
75  return out_it;
