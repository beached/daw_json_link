230  template<typename OutputIterator>
231  constexpr void utf32_to_utf8( std::uint32_t cp, OutputIterator &it ) {
232  if( cp <= 0x7FU ) {
233  *it++ = static_cast<char>( cp );
234  return;
235  }
236  if( cp <= 0x7FFU ) {
237  *it++ = static_cast<char>( ( cp >> 6U ) | 0b11000000U );
238  *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
239  return;
240  }
241  if( cp <= 0xFFFFU ) {
242  *it++ = static_cast<char>( ( cp >> 12U ) | 0b11100000U );
243  *it++ =
244  static_cast<char>( ( ( cp >> 6U ) & 0b00111111U ) | 0b10000000U );
245  *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
246  return;
247  }
248  if( cp <= 0x10FFFFU ) {
249  *it++ = static_cast<char>( ( cp >> 18U ) | 0b11110000U );
250  *it++ =
251  static_cast<char>( ( ( cp >> 12U ) & 0b00111111U ) | 0b10000000U );
252  *it++ =
253  static_cast<char>( ( ( cp >> 6U ) & 0b00111111U ) | 0b10000000U );
254  *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
255  return;
256  }
257  daw_json_error( ErrorReason::InvalidUTFCodepoint );
258  }

262  template<bool do_escape = false,
263  EightBitModes EightBitMode = EightBitModes::AllowFull,
264  typename OutputIterator, typename Container,
265  daw::enable_when_t<traits::is_container_like_v<

267  [[nodiscard]] constexpr OutputIterator
268  copy_to_iterator( OutputIterator it, Container const &container ) {
269  if constexpr( do_escape ) {
270  using iter = daw::remove_cvref_t<decltype( std::begin( container ) )>;
271  using it_t = utf8::unchecked::iterator<iter>;
272  auto first = it_t( std::begin( container ) );
273  auto const last = it_t( std::end( container ) );
274  while( first != last ) {
275  auto const cp = *first++;
276  switch( cp ) {
277  case '"':
278  *it++ = '\\';
279  *it++ = '"';
280  break;
281  case '\\':
282  *it++ = '\\';
283  *it++ = '\\';
284  break;
285  case '\b':
286  *it++ = '\\';
287  *it++ = 'b';
288  break;
289  case '\f':
290  *it++ = '\\';
291  *it++ = 'f';
292  break;
293  case '\n':
294  *it++ = '\\';
295  *it++ = 'n';
296  break;
297  case '\r':
298  *it++ = '\\';
299  *it++ = 'r';
300  break;
301  case '\t':
302  *it++ = '\\';
303  *it++ = 't';
304  break;
305  default:
306  if( cp < 0x20U ) {
307  it = json_details::output_hex( static_cast<std::uint16_t>( cp ),
308  it );
309  break;
310  }
311  if constexpr( EightBitMode == EightBitModes::DisallowHigh ) {
312  if( cp >= 0x7FU and cp <= 0xFFFFU ) {
313  it = json_details::output_hex(
314  static_cast<std::uint16_t>( cp ), it );
315  break;
316  }
317  if( cp > 0xFFFFU ) {
318  it = json_details::output_hex(
319  static_cast<std::uint16_t>( 0xD7C0U + ( cp >> 10U ) ), it );
320  it = json_details::output_hex(
321  static_cast<std::uint16_t>( 0xDC00U + ( cp & 0x3FFU ) ),
322  it );
323  break;
324  }
325  }
326  json_details::utf32_to_utf8( cp, it );
327  break;
328  }
329  }
330  } else {
331  for( auto c : container ) {
332  if constexpr( EightBitMode == EightBitModes::DisallowHigh ) {
333  daw_json_assert( ( static_cast<unsigned char>( c ) >= 0x20U and
334  static_cast<unsigned char>( c ) <= 0x7FU ),
335  ErrorReason::InvalidStringHighASCII );
336  }
337  *it++ = c;
338  }
339  }
340  return it;

643  LiteralAsStringOpt::Always ) {
644  *it++ = '"';

648  }
