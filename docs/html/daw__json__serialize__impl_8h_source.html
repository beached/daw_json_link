13  /***
14  * Serialize items to an output iterator as members of a class
15  * @tparam JsonMembers member items in json_class
16  * @tparam OutputIterator An Output Iterator that allows writing character
17  * data
18  * @tparam Is index_sequence index into JsonMembers
19  * @tparam Tuple tuple type holding class members
20  * @tparam Value mapped class type to serialize
21  * @param it an Output Iterator to write char data to
22  * @param args A tuple of the member values
23  * @param value class to serialize
24  * @return The OutputIterator it at the final position
25  */
26  template<typename... JsonMembers, typename OutputIterator, std::size_t... Is,
27  typename Tuple, typename Value>
28  [[nodiscard]] inline constexpr OutputIterator
29  serialize_json_class( OutputIterator it, std::index_sequence<Is...>,
30  Tuple const &args, Value const &value ) {
31 
32  bool is_first = true;
33  *it++ = '{';
34 
35  auto visited_members =
36  daw::bounded_vector_t<daw::string_view, sizeof...( JsonMembers ) * 2U>{ };
37  // Tag Members, if any. Putting them ahead means we can parse this faster
38  // in the future
39  (void)( ( tags_to_json_str<Is,
40  daw::traits::nth_element<Is, JsonMembers...>>(
41  is_first, it, value, visited_members ),
42  ... ),
43  0 );
44  // Regular Members
45  (void)( ( to_json_str<Is, daw::traits::nth_element<Is, JsonMembers...>>(
46  is_first, it, args, value, visited_members ),
47  ... ),
48  0 );
49 
50  *it++ = '}';
51  return it;
52  }
