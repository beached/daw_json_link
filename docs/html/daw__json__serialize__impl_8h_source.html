21  /***
22  * Serialize items to an output iterator as members of a class
23  * @tparam JsonMembers member items in json_class
24  * @tparam OutputIterator An Output Iterator that allows writing character
25  * data
26  * @tparam Is index_sequence index into JsonMembers
27  * @tparam Tuple tuple type holding class members
28  * @tparam Value mapped class type to serialize
29  * @param it an Output Iterator to write char data to
30  * @param args A tuple of the member values
31  * @param value class to serialize
32  * @return The OutputIterator it at the final position
33  */
34  template<typename... JsonMembers, typename OutputIterator,
35  std::size_t... Is, typename Tuple, typename Value>
36  [[nodiscard]] inline constexpr OutputIterator
37  serialize_json_class( OutputIterator it, std::index_sequence<Is...>,
38  Tuple const &args, Value const &value ) {
39 
40  bool is_first = true;
41  *it++ = '{';
42 
43  auto visited_members =
44  daw::bounded_vector_t<daw::string_view,
45  sizeof...( JsonMembers ) * 2U>{ };
46  // Tag Members, if any. Putting them ahead means we can parse this
47  // faster in the future
48  (void)( ( tags_to_json_str<Is, traits::nth_element<Is, JsonMembers...>>(
49  is_first, it, value, visited_members ),
50  ... ),
51  0 );
52  // Regular Members
53  (void)( ( to_json_str<Is, traits::nth_element<Is, JsonMembers...>>(
54  is_first, it, args, value, visited_members ),
55  ... ),
56  0 );
57 
58  *it++ = '}';
59  return it;
60  }
