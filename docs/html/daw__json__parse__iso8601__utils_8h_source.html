62  template<typename Result, typename Bounds, std::ptrdiff_t Ex>
63  constexpr Result
64  parse_number( daw::basic_string_view<char, Bounds, Ex> sv ) {
65  static_assert( daw::numeric_limits<Result>::digits10 >= 4 );
66  daw_json_assert( not sv.empty( ), "Invalid number" );
67  Result result = 0;
68  Result sign = 1;
69  if( sv.front( ) == '-' ) {
70  if constexpr( daw::is_signed_v<Result> ) {
71  sign = -1;
72  }
73  sv.remove_prefix( );
74  } else if( sv.front( ) == '+' ) {
75  sv.remove_prefix( );
76  }
77  while( not sv.empty( ) ) {
78  auto const dig =
79  static_cast<unsigned>( static_cast<unsigned char>( sv.pop_front( ) ) -
80  static_cast<unsigned char>( '0' ) );
81  daw_json_assert( dig < 10U, "Invalid digit" );
82  result *= 10;
83  result += static_cast<Result>( dig );
84  }
85  return result * sign;
86  }

89  // https://stackoverflow.com/questions/16773285/how-to-convert-stdchronotime-point-to-stdtm-without-using-time-t
90  template<typename Clock = std::chrono::system_clock,
91  typename Duration = std::chrono::milliseconds>
92  constexpr std::chrono::time_point<Clock, Duration>
93  civil_to_time_point( std::int_least32_t yr, std::uint_least32_t mo,
94  std::uint_least32_t dy, std::uint_least32_t hr,
95  std::uint_least32_t mn, std::uint_least32_t se,
96  std::uint_least32_t ms ) {
97  constexpr auto calc = []( std::int_least32_t y, std::uint_least32_t m,
98  std::uint_least32_t d, std::uint_least32_t h,
99  std::uint_least32_t min, std::uint_least32_t s,
100  std::uint_least32_t mil ) {
101  y -= static_cast<std::int_least32_t>( m ) <= 2;
102  std::int_least32_t const era = ( y >= 0 ? y : y - 399 ) / 400;
103  auto const yoe = static_cast<std::uint_least32_t>(
104  static_cast<std::int_least32_t>( y ) - era * 400 ); // [0, 399]
105  auto const doy = static_cast<std::uint_least32_t>(
106  ( 153 * ( static_cast<std::int_least32_t>( m ) +
107  ( static_cast<std::int_least32_t>( m ) > 2 ? -3 : 9 ) ) +
108  2 ) /
109  5 +
110  static_cast<std::int_least32_t>( d ) - 1 ); // [0, 365]
111  std::uint_least32_t const doe =
112  yoe * 365 + yoe / 4 - yoe / 100 + doy; // [0, 146096]
113  std::int_least32_t const days_since_epoch =
114  era * 146097 + static_cast<std::int_least32_t>( doe ) - 719468;
115 
116  using Days = std::chrono::duration<std::int_least32_t, std::ratio<86400>>;
117  return std::chrono::time_point<std::chrono::system_clock,
118  std::chrono::milliseconds>{ } +
119  ( Days( days_since_epoch ) + std::chrono::hours( h ) +
120  std::chrono::minutes( min ) +
121  std::chrono::seconds( static_cast<std::uint_least32_t>( s ) ) +
122  std::chrono::milliseconds( mil ) );
123  };
124  // Not all clocks have the same epoch. This should account for the offset
125  // and adjust the time_point so that the days prior are in relation to unix
126  // epoch. If system_clock is used, as is the default for the return value,
127  // it will be zero and should be removed by the compiler
128  auto result = calc( yr, mo, dy, hr, mn, se, ms );
129 #if defined( __cpp_lib_chrono ) and __cpp_lib_chrono >= 201907
130  // We have clock_cast
131  return std::chrono::duration_cast<Duration>(
132  std::chrono::clock_cast<Clock>( result ) );
133 #else
134  if constexpr( std::is_same_v<Clock, std::chrono::system_clock> ) {
135  return result;
136  } else {
137  // This is a guess and will not be constexpr
138 
139  // System epoch is unix epoch on(gcc/clang/msvc)
140  auto const system_epoch = std::chrono::floor<std::chrono::hours>(
141  std::chrono::system_clock::now( ).time_since_epoch( ) +
142  std::chrono::minutes( 30 ) );
143  auto const clock_epoch = std::chrono::floor<std::chrono::hours>(
144  Clock::now( ).time_since_epoch( ) + std::chrono::minutes( 30 ) );
145 
146  constexpr auto offset =
147  std::chrono::duration_cast<std::chrono::milliseconds>( clock_epoch -
148  system_epoch );
149  return std::chrono::duration_cast<Duration>( result + offset );
150  }
151 #endif
152  }

187  constexpr time_parts parse_iso_8601_time(
188  daw::basic_string_view<char, Bounds, Ex> timestamp_str ) {
189  auto result = time_parts{ 0, 0, 0, 0 };
190  result.hour = parse_utils::parse_unsigned<std::uint_least32_t, 2>(
191  timestamp_str.pop_front( 2 ).data( ) );
192  if( not parse_utils::is_number( timestamp_str.front( ) ) ) {
193  timestamp_str.remove_prefix( );
194  }
195  result.minute = parse_utils::parse_unsigned<std::uint_least32_t, 2>(
196  timestamp_str.pop_front( 2 ).data( ) );
197  if( timestamp_str.empty( ) ) {
198  return result;
199  }
200  if( not parse_utils::is_number( timestamp_str.front( ) ) ) {
201  timestamp_str.remove_prefix( );
202  }
203  result.second = parse_utils::parse_unsigned<std::uint_least32_t, 2>(
204  timestamp_str.pop_front( 2 ).data( ) );
205  if( timestamp_str.empty( ) ) {
206  return result;

216  template<typename Bounds, std::ptrdiff_t Ex>
217  constexpr std::chrono::time_point<std::chrono::system_clock,
218  std::chrono::milliseconds>
219  parse_iso8601_timestamp( daw::basic_string_view<char, Bounds, Ex> ts ) {
220  constexpr daw::string_view t_str = "T";
221  auto const date_str = ts.pop_front( t_str );
222  if( ts.empty( ) ) {
223  daw_json_error( "Invalid timestamp, missing T separator" );
224  }
225  date_parts const ymd = parse_iso_8601_date( date_str );
226  auto time_str = ts.pop_front( []( char c ) {
227  return not( parse_utils::is_number( c ) | (c == ':') | (c == '.') );
228  } );
229  // TODO: verify or parse timezone
230  time_parts hms = parse_iso_8601_time( time_str );
231  if( not( ts.empty( ) or ts.front( ) == 'Z' ) ) {
232  daw_json_assert( ts.size( ) == 5 or ts.size( ) == 6,
233  "Invalid timezone offset" );
234  // The format will be (+|-)hh[:]mm
235  bool const sign = ts.front( ) == '+';
236  ts.remove_prefix( );
237  auto hr_offset =
238  parse_utils::parse_unsigned<std::uint_least32_t, 2>( ts.data( ) );
239  if( ts.front( ) == ':' ) {
240  ts.remove_prefix( );
241  }
242  auto mn_offset =
243  parse_utils::parse_unsigned<std::uint_least32_t, 2>( ts.data( ) );
244  // Want to subtract offset from current time, we are converting to UTC
245  if( sign ) {
246  // Positive offset
247  hms.hour -= hr_offset;
248  hms.minute -= mn_offset;
249  } else {
250  // Negative offset
251  hms.hour += hr_offset;
252  hms.minute += mn_offset;
253  }
254  }
255  return civil_to_time_point( ymd.year, ymd.month, ymd.day, hms.hour,
256  hms.minute, hms.second, hms.millisecond );
257  }

268  template<typename Clock, typename Duration>
269  constexpr ymdhms time_point_to_civil(
270  std::chrono::time_point<Clock, Duration> const &tp ) noexcept {
271  auto dur_from_epoch = tp.time_since_epoch( );
272  using Days = std::chrono::duration<std::int_least32_t, std::ratio<86400>>;
273  auto const days_since_epoch =
274  std::chrono::duration_cast<Days>( dur_from_epoch );
275  std::int_least32_t z = days_since_epoch.count( );
276  z += 719468;
277  std::int_least32_t const era = ( z >= 0 ? z : z - 146096 ) / 146097;
278  auto const doe =
279  static_cast<std::uint_least32_t>( z - era * 146097 ); // [0, 146096]
280  std::uint_least32_t const yoe =
281  ( doe - doe / 1460 + doe / 36524 - doe / 146096 ) / 365; // [0, 399]
282  std::int_least32_t const y =
283  static_cast<std::int_least32_t>( yoe ) + era * 400;
284  std::uint_least32_t const doy =
285  doe - ( 365 * yoe + yoe / 4 - yoe / 100 ); // [0, 365]
286  std::uint_least32_t const mp = ( 5 * doy + 2 ) / 153; // [0, 11]
287  std::uint_least32_t const d = doy - ( 153 * mp + 2 ) / 5 + 1; // [1, 31]
288  auto const m = static_cast<std::uint_least32_t>(
289  static_cast<std::int_least32_t>( mp ) +
290  ( static_cast<std::int_least32_t>( mp ) < 10 ? 3 : -9 ) ); // [1, 12]
291 
292  dur_from_epoch -= days_since_epoch;
293  auto const hrs =
294  std::chrono::duration_cast<std::chrono::hours>( dur_from_epoch );
295  dur_from_epoch -= hrs;
296  auto const min =
297  std::chrono::duration_cast<std::chrono::minutes>( dur_from_epoch );
298  dur_from_epoch -= min;
299  auto const sec =
300  std::chrono::duration_cast<std::chrono::seconds>( dur_from_epoch );
301  dur_from_epoch -= sec;
302  auto const ms =
303  std::chrono::duration_cast<std::chrono::milliseconds>( dur_from_epoch );
304  return ymdhms{ y + ( m <= 2 ),
305  m,
306  d,
307  static_cast<std::uint_least32_t>( hrs.count( ) ),
308  static_cast<std::uint_least32_t>( min.count( ) ),
309  static_cast<std::uint_least32_t>( sec.count( ) ),
310  static_cast<std::uint_least32_t>( ms.count( ) ) };
311  }
