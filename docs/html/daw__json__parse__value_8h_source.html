223  parse_value( ParseTag<JsonParseTypes::Bool>, Range &rng ) {

255  rng.remove_prefix( 5 );

257  } else {
258  if( rng.starts_with( "true" ) ) {
259  rng.remove_prefix( 4 );
260  result = true;
261  } else if( rng.starts_with( "false" ) ) {
262  rng.remove_prefix( 5 );
263  } else {

265  }
266  }
267  if constexpr( JsonMember::literal_as_string !=
268  LiteralAsStringOpt::Never ) {
269  skip_quote_when_literal_as_string<JsonMember::literal_as_string>( rng );
270  }
271  rng.trim_left( );
272  daw_json_assert_weak(
273  parse_policy_details::at_end_of_item( rng.front( ) ),
274  ErrorReason::InvalidEndOfValue, rng );
