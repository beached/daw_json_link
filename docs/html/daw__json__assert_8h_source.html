284  /***
285  * Helper to provide output formatted information about json_exception
286  * @param je json_exception to be formatted
287  * @return string representation of json_exception
288  */
289  inline std::string
290  to_formatted_string( json_exception const &je,
291  char const *json_document = nullptr ) {
292  using namespace std::string_literals;
293  std::string result = "reason: "s + je.reason( );
294  if( json_document == nullptr or je.parse_location( ) == nullptr ) {
295  return result;
296  }
297  auto const previous_char_count =
298  std::min( static_cast<std::size_t>( 100 ),
299  static_cast<std::size_t>(
300  std::distance( json_document, je.parse_location( ) ) ) );
301  auto const loc_data = std::string_view(
302  std::prev( je.parse_location( ),
303  static_cast<std::ptrdiff_t>( previous_char_count ) ),
304  previous_char_count );
305 #ifndef _WIN32
306  result += "\nlocation:\x1b[1m";
307 #endif
308  result +=
309  std::accumulate( loc_data.data( ), loc_data.data( ) + loc_data.size( ),
310  std::string{ }, []( std::string s, char c ) {
311  if( ( c != '\n' ) & ( c != '\r' ) ) {
312  s += c;
313  }
314  return s;
315  } );
316 #ifndef _WIN32
317  result += "\x1b[0m\n";
318 #endif
319  return result;
320  }

347 daw_json_error( daw::json::ErrorReason reason, Range const &location ) {
348 #ifdef DAW_USE_JSON_EXCEPTIONS
349  if constexpr( ShouldThrow ) {
350  if( location.first ) {
351  if( location.class_first ) {
352  auto const len =
353  static_cast<std::size_t>( location.first - location.class_first );
354  auto loc = std::string_view( location.class_first, len );
355  throw daw::json::json_exception( reason, loc );
356  }
357  if( location.last ) {
358  auto const len =
359  static_cast<std::size_t>( location.last - location.first );
360  auto loc = std::string_view( location.first, len );
361  throw daw::json::json_exception( reason, loc );
362  }
363  throw daw::json::json_exception( reason,
364  std::string_view( location.first, 0 ) );
365  } else {
