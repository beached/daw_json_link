81  [[nodiscard]] inline std::string
82  to_json_path_string( std::vector<json_path_node> const &path_stack ) {
83  return daw::algorithm::accumulate(
84  std::data( path_stack ), daw::data_end( path_stack ), std::string{ },
85  []( auto &&state, json_path_node const &sv ) mutable {
86  if( sv.index( ) >= 0 ) {
87  state += "[" + std::to_string( sv.index( ) ) + "]";
88  } else if( not sv.name( ).empty( ) ) {
89  state += "." + static_cast<std::string>( sv.name( ) );
90  }
91  return DAW_FWD( state );
92  } );

126  [[nodiscard]] bool handle_on_value( json_pair jp ) {
127  if( auto const range = jp.value.get_raw_state( );
128  range.empty( ) or last <= std::data( range ) ) {
129  return false;
130  }
131  if( auto const t = child_of( ); t == JsonBaseParseTypes::Class ) {
132  state.m_name = *jp.name;
133  state.m_index = -1;
134  } else if( t == JsonBaseParseTypes::Array ) {
135  state.m_name = { };
136  state.m_index++;
137  } else {
138  state.m_name = { };
139  state.m_index = -1;
140  }
141  state.m_value_start = jp.value.get_raw_state( ).first;
142  state.m_type = jp.value.type( );

288  [[nodiscard]] constexpr std::size_t
289  find_line_number_of( char const *doc_pos, char const *doc_start ) {
290  daw_json_ensure( doc_pos != nullptr and doc_start != nullptr,
291  ErrorReason::UnexpectedEndOfData );
292  daw_json_ensure( std::less<>{ }( doc_start, doc_pos ),
293  ErrorReason::UnexpectedEndOfData );
294 
295  return daw::algorithm::accumulate( doc_start, doc_pos, std::size_t{ },
296  []( std::size_t count, char c ) {
297  if( c == '\n' ) {
298  return count + 1;
299  }
300  return count;
301  } );

309  [[nodiscard]] constexpr std::size_t
310  find_column_number_of( char const *doc_pos, char const *doc_start ) {
311  daw_json_ensure( doc_pos != nullptr and doc_start != nullptr,
312  ErrorReason::UnexpectedEndOfData );
313  daw_json_ensure( std::less<>{ }( doc_start, doc_pos ),
314  ErrorReason::UnexpectedEndOfData );
315 
316  auto first = daw::reverse_iterator<char const *>( doc_pos );
317  auto last = daw::reverse_iterator<char const *>( doc_start );
318  auto pos =
319  std::distance( first, daw::algorithm::find( first, last, '\n' ) );
320  daw_json_ensure( pos >= 0, ErrorReason::Unknown );
321  return static_cast<std::size_t>( pos );
322  }
