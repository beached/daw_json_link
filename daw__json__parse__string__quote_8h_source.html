124  template<typename Range>
125  [[nodiscard]] static constexpr auto parse_nq( Range &rng )
126  -> std::enable_if_t<not Range::is_unchecked_input, std::size_t> {
127  std::ptrdiff_t need_slow_path = -1;
128  char const *first = rng.first;
129  char const *const last = rng.class_last;
130  if constexpr( not std::is_same_v<typename Range::exec_tag_t,
131  constexpr_exec_tag> ) {
132  first = mem_skip_until_end_of_string<false>( Range::exec_tag, first,
133  last, need_slow_path );
134  } else {
135  if( char const *const l = rng.last; l - first >= 8 ) {
136  skip_to_first8( first, l );
137  } else if( last - first >= 4 ) {
138  skip_to_first4( first, l );
139  }
140  while( DAW_JSON_LIKELY( first < last ) and
141  DAW_JSON_LIKELY( *first != '"' ) ) {
142  while( DAW_JSON_LIKELY( first < last ) and
143  ( DAW_JSON_LIKELY( *first != '"' ) &
144  DAW_JSON_LIKELY( *first != '\\' ) ) ) {
145  ++first;
146  }
147 
148  if( DAW_JSON_UNLIKELY( first < last and *first == '\\' ) ) {
149  if( need_slow_path < 0 ) {
150  need_slow_path = first - rng.first;
151  }
152  first += 2;
153  } else {
154  break;
155  }
156  }
157  }
158  daw_json_assert_weak( first < last and *first == '"',
159  ErrorReason::InvalidString, rng );
160  rng.first = first;
