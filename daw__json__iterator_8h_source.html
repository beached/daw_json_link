60  /***
61  * Iterator for iterating over JSON array's
62  * @tparam JsonElement type under underlying element in array. If
63  * heterogeneous, a basic_json_value_iterator may be more appropriate

67  class json_array_iterator {

72  static inline constexpr ParseState
73  get_range( daw::string_view data, daw::string_view member_path ) {
74  auto [is_found, result] = json_details::find_range<ParseState>(
75  DAW_FWD( data ),
76  { std::data( member_path ), std::size( member_path ) } );
77  daw_json_ensure( is_found, ErrorReason::JSONPathNotFound );
78  daw_json_ensure( result.front( ) == '[', ErrorReason::InvalidArrayStart,
79  result );
80  return result;
81  }

131  [[nodiscard]] inline constexpr value_type operator*( ) const {
132  daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
133  ErrorReason::UnexpectedEndOfData, m_state );
134 
135  auto tmp = m_state;
136 
137  if constexpr( is_pinned_type_v<typename element_type::parse_to_t> ) {
138  auto const run_after_parse =
139  json_array_iterator_details::op_star_cleanup<CharT, ParseState>{
140  m_can_skip, tmp };
141  (void)run_after_parse;
142  return json_details::parse_value<element_type>(
143  tmp, ParseTag<element_type::expected_type>{ } );
144  } else {
145  auto result = json_details::parse_value<element_type>(
146  tmp, ParseTag<element_type::expected_type>{ } );
147 
148  m_can_skip = tmp.first;
149  return result;
150  }
