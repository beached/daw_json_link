60  /***
61  * Iterator for iterating over JSON array's
62  * @tparam JsonElement type under underlying element in array. If
63  * heterogeneous, a basic_json_value_iterator may be more appropriate
64  * @tparam ParsePolicy Parsing policy type
65  */
66  template<typename JsonElement, auto... PolicyFlags>
67  class json_array_iterator {
68  using ParseState = BasicParsePolicy<
69  options::details::make_parse_flags<PolicyFlags...>( ).value>;
70  using CharT = typename ParseState::CharT;
71 
72  static inline constexpr ParseState
73  get_range( daw::string_view data, daw::string_view member_path ) {
74  auto [is_found, result] = json_details::find_range<ParseState>(
75  DAW_FWD( data ),
76  { std::data( member_path ), std::size( member_path ) } );
77  daw_json_assert( is_found, ErrorReason::JSONPathNotFound );
78  daw_json_assert( result.front( ) == '[', ErrorReason::InvalidArrayStart,
79  result );
80  return result;
81  }

83  public:
84  using element_type = json_details::json_deduced_type<JsonElement>;
85  static_assert( not std::is_same_v<element_type, void>,
86  "Unknown JsonElement type." );
87  using value_type = typename element_type::parse_to_t;
88  using reference = value_type;

128  /// @brief Parse the current element
129  /// @pre good( ) returns true
130  /// @return The parsed result of ParseElement
131  [[nodiscard]] inline constexpr value_type operator*( ) const {
132  daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
133  ErrorReason::UnexpectedEndOfData, m_state );
134 
135  auto tmp = m_state;
136 
137  if constexpr( json_details::use_direct_construction_v<ParseState,
138  JsonElement> ) {
139  auto const run_after_parse =
140  json_array_iterator_details::op_star_cleanup<CharT, ParseState>{
141  m_can_skip, tmp };
142  (void)run_after_parse;
143  return json_details::parse_value<element_type>(
144  tmp, ParseTag<element_type::expected_type>{ } );
145  } else {
146  auto result = json_details::parse_value<element_type>(
147  tmp, ParseTag<element_type::expected_type>{ } );
148 
149  m_can_skip = tmp.first;
150  return result;
151  }
152  }

163  /***
164  * Move the parse state to the next element
165  * @return iterator after moving
166  */
167  inline constexpr json_array_iterator &operator++( ) {
168  daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
169  ErrorReason::UnexpectedEndOfData, m_state );
170  if( m_can_skip ) {
171  m_state.first = m_can_skip;
172  m_can_skip = nullptr;
173  } else {
174  (void)json_details::skip_known_value<element_type>( m_state );
175  }
176  m_state.move_next_member_or_end( );
177  return *this;
178  }
