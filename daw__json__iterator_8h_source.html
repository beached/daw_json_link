117  /***
118  * Parse the current element
119  * @pre good( ) returns true
120  * @return The parsed result of ParseElement
121  */
122  [[nodiscard]] inline constexpr value_type operator*( ) const {
123  daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
124  ErrorReason::UnexpectedEndOfData, m_state );
125 
126  auto tmp = m_state;
127 
128  if constexpr( json_details::is_guaranteed_rvo_v<ParsePolicy> ) {
129  auto const ae = daw::on_exit_success( [&] { m_can_skip = tmp.first; } );
130  return json_details::parse_value<element_type>(
131  ParseTag<element_type::expected_type>{ }, tmp );
132  } else {
133  auto result = json_details::parse_value<element_type>(
134  ParseTag<element_type::expected_type>{ }, tmp );
135 
136  m_can_skip = tmp.first;
137  return result;
138  }
139  }
