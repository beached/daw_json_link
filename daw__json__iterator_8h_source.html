53  /***
54  * Iterator for iterating over JSON array's
55  * @tparam JsonElement type under underlying element in array. If
56  * heterogeneous, a basic_json_value_iterator may be more appropriate
57  * @tparam ParsePolicy Parsing policy type
58  */
59  template<typename JsonElement, auto... PolicyFlags>
60  class json_array_iterator {
61  using ParseState = TryDefaultParsePolicy<BasicParsePolicy<
62  options::details::make_parse_flags<PolicyFlags...>( ).value>>;
63  using CharT = typename ParseState::CharT;
64 
65  static inline constexpr ParseState
66  get_range( daw::string_view data, daw::string_view member_path ) {
67  auto [is_found, result] = json_details::find_range<ParseState>(
68  DAW_FWD( data ),
69  { std::data( member_path ), std::size( member_path ) } );
70  daw_json_ensure( is_found, ErrorReason::JSONPathNotFound );
71  daw_json_ensure( result.front( ) == '[', ErrorReason::InvalidArrayStart,
72  result );
73  return result;
74  }
75 
76  public:
77  using element_type = json_details::json_deduced_type<JsonElement>;
78  static_assert( not std::is_same_v<element_type, void>,
79  "Unknown JsonElement type." );
80  using value_type = typename element_type::parse_to_t;
81  using reference = value_type;
82  using pointer = json_details::arrow_proxy<value_type>;
83  using difference_type = std::ptrdiff_t;
84  // Can do forward iteration and be stored
85  using iterator_category = std::input_iterator_tag;
86 
87  private:
88  ParseState m_state = ParseState( );
89  /***
90  * This lets us fastpath and just skip n characters as we have already
91  * parsed them
92  */
93  mutable CharT *m_can_skip = nullptr;
94 
95  public:
96  json_array_iterator( ) = default;
97 
98  inline constexpr explicit json_array_iterator( daw::string_view jd )
99  : m_state( ParseState( std::data( jd ), daw::data_end( jd ) ) ) {
100 
101  m_state.trim_left( );
102  daw_json_assert_weak( m_state.is_opening_bracket_checked( ),
103  ErrorReason::InvalidArrayStart, m_state );
104 
105  m_state.remove_prefix( );
106  m_state.trim_left( );
107  }
108 
109  inline constexpr explicit json_array_iterator(
110  daw::string_view jd, daw::string_view start_path )
111  : m_state( get_range( jd, start_path ) ) {
112 
113  m_state.trim_left( );
114  daw_json_assert_weak( m_state.is_opening_bracket_checked( ),
115  ErrorReason::InvalidArrayStart, m_state );
116 
117  m_state.remove_prefix( );
118  m_state.trim_left( );
119  }
120 
124  [[nodiscard]] inline constexpr value_type operator*( ) const {
125  daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
126  ErrorReason::UnexpectedEndOfData, m_state );
127 
128  auto tmp = m_state;
129 
130  auto const run_after_parse =
131  json_details::assign_on_dtor{ m_can_skip, tmp.first };
132  (void)run_after_parse;
133  return json_details::parse_value<element_type>(
134  tmp, ParseTag<element_type::expected_type>{ } );
135  }

142  [[nodiscard]] inline pointer operator->( ) const {

144  }

146  /***
147  * Move the parse state to the next element
148  * @return iterator after moving
149  */
150  inline constexpr json_array_iterator &operator++( ) {
151  daw_json_assert_weak( m_state.has_more( ) and m_state.front( ) != ']',
152  ErrorReason::UnexpectedEndOfData, m_state );
153  if( m_can_skip ) {
154  m_state.first = m_can_skip;
155  m_can_skip = nullptr;
156  } else {
157  (void)json_details::skip_known_value<element_type>( m_state );
158  }
159  m_state.move_next_member_or_end( );
160  return *this;
161  }

174  [[nodiscard]] inline constexpr bool good( ) const {

190  if( not( *this ) ) {
191  return not rhs;

197  }

202  [[nodiscard]] inline constexpr bool
203  operator!=( json_array_iterator const &rhs ) const {
204  if( not( *this ) ) {
205  return static_cast<bool>( rhs );
206  }
207  if( not rhs ) {
208  return true;
209  }
210  return m_state.first != rhs.m_state.first;
211  }
