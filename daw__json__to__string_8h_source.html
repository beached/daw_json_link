211  template<typename OutputIterator>
212  constexpr void utf32_to_utf8( std::uint32_t cp, OutputIterator &it ) {
213  if( cp <= 0x7FU ) {
214  *it++ = static_cast<char>( cp );
215  return;
216  }
217  if( cp <= 0x7FFU ) {
218  *it++ = static_cast<char>( ( cp >> 6U ) | 0b11000000U );
219  *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
220  return;
221  }
222  if( cp <= 0xFFFFU ) {
223  *it++ = static_cast<char>( ( cp >> 12U ) | 0b11100000U );
224  *it++ = static_cast<char>( ( ( cp >> 6U ) & 0b00111111U ) | 0b10000000U );
225  *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
226  return;
227  }
228  if( cp <= 0x10FFFFU ) {
229  *it++ = static_cast<char>( ( cp >> 18U ) | 0b11110000U );
230  *it++ =
231  static_cast<char>( ( ( cp >> 12U ) & 0b00111111U ) | 0b10000000U );
232  *it++ = static_cast<char>( ( ( cp >> 6U ) & 0b00111111U ) | 0b10000000U );
233  *it++ = static_cast<char>( ( cp & 0b00111111U ) | 0b10000000U );
234  return;
235  }
236  daw_json_error( ErrorReason::InvalidUTFCodepoint );
237  }

241  template<bool do_escape = false,
242  EightBitModes EightBitMode = EightBitModes::AllowFull,
243  typename OutputIterator, typename Container,
244  daw::enable_when_t<daw::traits::is_container_like_v<
245  daw::remove_cvref_t<Container>>> = nullptr>
246  [[nodiscard]] constexpr OutputIterator
247  copy_to_iterator( OutputIterator it, Container const &container ) {
248  if constexpr( do_escape ) {
249  using iter = daw::remove_cvref_t<decltype( std::begin( container ) )>;
250  using it_t = utf8::unchecked::iterator<iter>;
251  auto first = it_t( std::begin( container ) );
252  auto const last = it_t( std::end( container ) );
253  while( first != last ) {
254  auto const cp = *first++;
255  switch( cp ) {
256  case '"':
257  *it++ = '\\';
258  *it++ = '"';
259  break;
260  case '\\':
261  *it++ = '\\';
262  *it++ = '\\';
263  break;
264  case '\b':
265  *it++ = '\\';
266  *it++ = 'b';
267  break;
268  case '\f':
269  *it++ = '\\';
270  *it++ = 'f';
271  break;
272  case '\n':
273  *it++ = '\\';
274  *it++ = 'n';
275  break;
276  case '\r':
277  *it++ = '\\';
278  *it++ = 'r';
279  break;
280  case '\t':
281  *it++ = '\\';
282  *it++ = 't';
283  break;
284  default:
285  if( cp < 0x20U ) {
286  it =
287  json_details::output_hex( static_cast<std::uint16_t>( cp ), it );
288  break;
289  }
290  if constexpr( EightBitMode == EightBitModes::DisallowHigh ) {
291  if( cp >= 0x7FU and cp <= 0xFFFFU ) {
292  it = json_details::output_hex( static_cast<std::uint16_t>( cp ),
293  it );
294  break;
295  }
296  if( cp > 0xFFFFU ) {
297  it = json_details::output_hex(
298  static_cast<std::uint16_t>( 0xD7C0U + ( cp >> 10U ) ), it );
299  it = json_details::output_hex(
300  static_cast<std::uint16_t>( 0xDC00U + ( cp & 0x3FFU ) ), it );
301  break;
302  }
303  }
304  json_details::utf32_to_utf8( cp, it );
305  break;
306  }
307  }
308  } else {
309  for( auto c : container ) {
310  if constexpr( EightBitMode == EightBitModes::DisallowHigh ) {
311  daw_json_assert( ( static_cast<unsigned char>( c ) >= 0x20U and
312  static_cast<unsigned char>( c ) <= 0x7FU ),
313  ErrorReason::InvalidStringHighASCII );
314  }
315  *it++ = c;
316  }
317  }
318  return it;
319  }

581  template<typename JsonMember, typename OutputIterator, typename parse_to_t>
582  [[nodiscard]] constexpr OutputIterator
583  to_string( ParseTag<JsonParseTypes::Signed>, OutputIterator it,
584  parse_to_t const &value ) {
585 
586  static_assert(
587  std::is_convertible_v<parse_to_t, typename JsonMember::parse_to_t>,
588  "value must be convertible to specified type in class contract" );
589 
590  using std::to_string;
591  using to_strings::to_string;
592  using under_type = base_int_type_t<parse_to_t>;
593 
594  if constexpr( JsonMember::literal_as_string ==
595  LiteralAsStringOpt::Always ) {
596  *it++ = '"';
597  }
598  if constexpr( std::is_enum_v<parse_to_t> or
599  daw::is_integral_v<parse_to_t> ) {
600  auto v = static_cast<under_type>( value );
601 
602  char buff[daw::numeric_limits<under_type>::digits10 + 1]{ };
603  char *ptr = buff;
604  if( v < 0 ) {
605  *it++ = '-';
606  // Do 1 round here just incase we are
607  // daw::numeric_limits<intmax_t>::min( ) and cannot negate
608  *ptr++ = '0' - static_cast<char>( v % 10 );
609  v /= -10;
610  if( v == 0 ) {
611  *it++ = buff[0];
612  if constexpr( JsonMember::literal_as_string ==
613  LiteralAsStringOpt::Always ) {
614  *it++ = '"';
615  }
616  return it;
617  }
618  }
619  if( v == 0 ) {
620  *ptr++ = '0';
621  }
622  while( v >= 10 ) {
623  auto const tmp = static_cast<std::size_t>( v % 100 );
624  v /= 100;
625  ptr[0] = digits100[tmp][0];
626  ptr[1] = digits100[tmp][1];
627  ptr += 2;
628  }
629  if( v > 0 ) {
630  *ptr++ = '0' + static_cast<char>( v );
631  }
632  --ptr;
633  *it++ = *ptr;
634  while( ptr != buff ) {
635  --ptr;
636  *it++ = *ptr;
637  }
