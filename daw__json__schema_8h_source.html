274  template<typename WritableType, typename... JsonMembers>
275  struct json_class_processor<WritableType,
276  json_member_list<JsonMembers...>> {
277 
278  static constexpr std::size_t size = sizeof...( JsonMembers );

284  static constexpr WritableType process( WritableType out_it ) {
285  out_it = utils::output_kv( out_it, R"("type")", R"("object")" );
286  out_it.put( ',' );
287  out_it.next_member( );
288  out_it = utils::output_kv( out_it, R"("properties")", "{" );
289  if constexpr( sizeof...( JsonMembers ) > 0 ) {
290  out_it.add_indent( );
291  out_it.next_member( );
292  out_it = output_member_types(
293  out_it, std::index_sequence_for<JsonMembers...>{ } );
294  out_it.del_indent( );
295  out_it.next_member( );
296  }
297  out_it.put( '}' );
298  out_it.put( ',' );
299  out_it.next_member( );
300  out_it = utils::output_kv( out_it, R"("required")", "[" );
301  if constexpr( not is_empty_pack_v<JsonMembers...> ) {
302  out_it.add_indent( );
303  out_it = output_required_members( out_it );
304  out_it.del_indent( );
305  out_it.next_member( );
306  }
307  out_it.put( ']' );
308  if constexpr( ( has_dependent_member_v<JsonMembers> or ... ) ) {
309  out_it.put( ',' );
310  out_it.next_member( );
311  out_it = utils::output_kv( out_it, R"("dependencies")", "{" );
312  out_it.add_indent( );
313  bool is_first = true;
314  out_it = static_cast<WritableType>(
315  ( output_dependency<
316  json_link_no_name<json_link_no_name<JsonMembers>>>(
317  out_it, is_first ),
318  ... ) );
319  out_it.del_indent( );
320  if( not is_first ) {
321  // If we have at least 1 dependent member, is_first will be false
322  out_it.next_member( );
323  }
324  out_it.put( '}' );
325  }
326  return out_it;
327  }

339  template<typename JsonMember, std::size_t Idx>
340  static constexpr WritableType output_member_type( WritableType &out_it,
341  bool *seen ) {
342  if( seen[Idx] ) {
343  return out_it;
344  } else {
345  seen[Idx] = true;

349  out_it = to_json_schema<JsonMember>(
350  ParseTag<JsonMember::expected_type>{ }, out_it );
351  if constexpr( Idx + 1 < sizeof...( JsonMembers ) ) {
352  out_it.put( ',' );
353  out_it.next_member( );
354  }
355  return out_it;
