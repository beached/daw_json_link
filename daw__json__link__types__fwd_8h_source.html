384  template<JSONNAMETYPE Name, typename T>
385  DAW_ATTRIBUTE_HIDDEN inline constexpr auto json_link_quick_map( ) {
386  if constexpr( has_basic_type_map_v<T> ) {
387  constexpr auto mapped_type = json_link_basic_type_map<T>::parse_type;
388  if constexpr( mapped_type == JsonParseTypes::StringRaw ) {
389  return json_link_quick_map_type<json_string_raw<Name, T>>{ };
390  } else if constexpr( mapped_type == JsonParseTypes::StringEscaped ) {
391  return json_link_quick_map_type<json_string<Name, T>>{ };
392  } else if constexpr( mapped_type == JsonParseTypes::Bool ) {
393  return json_link_quick_map_type<json_bool<Name, T>>{ };
394  } else if constexpr( mapped_type == JsonParseTypes::Signed ) {
395  return json_link_quick_map_type<json_number<Name, T>>{ };
396  } else if constexpr( mapped_type == JsonParseTypes::Unsigned ) {
397  return json_link_quick_map_type<json_number<Name, T>>{ };
398  } else if constexpr( mapped_type == JsonParseTypes::Real ) {
399  return json_link_quick_map_type<json_number<Name, T>>{ };
400  } else {
401  return json_link_quick_map_type<void, false>{ };
402  }
403  } else {
404  return json_link_quick_map_type<void, false>{ };
405  }
406  }
