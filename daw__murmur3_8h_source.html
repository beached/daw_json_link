61  murmur3_32( StringView &&key, std::uint32_t seed = 0 ) {
62  UInt32 h = to_uint32( seed );
63  UInt32 k = 0_u32;
64  char const *first = std::data( key );
65  char const *const last = daw::data_end( key );
66  while( ( last - first ) >= 4 ) {
67  // Here is a source of differing results across endiannesses.
68  // A swap here has no effects on hash properties though.
69  k = daw::to_uint32_buffer( first );
70  first += 4;
71  h ^= murmur3_details::murmur3_32_scramble( k );
72  h = rotate_left<13>( h );

74  }
