143  template<bool skip_end_check, typename CharT>
144  DAW_ATTRIB_FLATINLINE [[nodiscard]] static inline constexpr CharT *
145  skip_digits( CharT *first, CharT *const last ) {
146  (void)last; // only used inside if constexpr and gcc9 warns
147  unsigned dig = parse_digit( *first );
148  while( dig < 10 ) {
149  ++first;
150  if constexpr( not skip_end_check ) {
151  if( DAW_UNLIKELY( first >= last ) ) {
152  break;
153  }
154  }
155  dig = parse_digit( *first );
156  }
157  return first;
158  }

219  [[nodiscard]] static constexpr ParseState
220  skip_number( ParseState &parse_state ) {
221  using CharT = typename ParseState::CharT;
222  daw_json_assert_weak( parse_state.has_more( ),
223  ErrorReason::UnexpectedEndOfData, parse_state );
224 
225  auto result = parse_state;
226  CharT *first = parse_state.first;
227  CharT *const last = parse_state.last;
228  if constexpr( ParseState::allow_leading_zero_plus ) {
229  if( *first == '-' ) {
230  ++first;
231  }
232  } else {
233  switch( *first ) {
234  case '-':
235  ++first;
236  break;
237  case '+':
238  daw_json_error( ErrorReason::InvalidNumberStart, parse_state );
239  case '0':
240  if( last - first > 1 ) {
241  daw_json_ensure(
242  not parse_policy_details::is_number( *std::next( first ) ),
243  ErrorReason::InvalidNumberStart, parse_state );
244  }
245  break;
246  }
247  }
248 
249  if( DAW_LIKELY( first < last ) ) {
250  DAW_LIKELY_BRANCH
251  first =
252  skip_digits<( ParseState::is_zero_terminated_string or
253  ParseState::is_unchecked_input )>( first, last );
254  }
255 
256  CharT *decimal = nullptr;
257  if( ( ( ParseState::is_zero_terminated_string or
258  ParseState::is_unchecked_input ) or
259  first < last ) and
260  ( *first == '.' ) ) {
261  decimal = first;
262  ++first;
263  if( DAW_LIKELY( first < last ) ) {
264  DAW_LIKELY_BRANCH
265  first =
266  skip_digits<( ParseState::is_zero_terminated_string or
267  ParseState::is_unchecked_input )>( first, last );
268  }
269  }
270  CharT *exp = nullptr;
271 
272  unsigned dig = [&] {
273  if( ParseState::is_zero_terminated_string or first < last ) {
274  return parse_digit( *first );
275  }
276  // We are out of range and the exponent part is optional
277  return 0U;
278  }( );
279  if( ( dig == parsed_constants::e_char ) |
280  ( dig == parsed_constants::E_char ) ) {
281  exp = first;
282  ++first;
283  daw_json_assert_weak( first < last, ErrorReason::UnexpectedEndOfData,
284  [&] {
285  auto r = parse_state;
286  r.first = first;
287  return r;
288  }( ) );
289  dig = parse_digit( *first );
290  if( ( dig == parsed_constants::plus_char ) |
291  ( dig == parsed_constants::minus_char ) ) {
292  ++first;
293  }
294  daw_json_assert_weak( first < last and parse_digit( *first ) < 10U,
295  ErrorReason::InvalidNumber );
296 
297  if( DAW_LIKELY( first < last ) ) {
298  DAW_LIKELY_BRANCH
299  first =
300  skip_digits<( ParseState::is_zero_terminated_string or
301  ParseState::is_unchecked_input )>( first, last );
302  }
303  }
304 
305  parse_state.first = first;
306  result.last = first;
307  result.class_first = decimal;
308  result.class_last = exp;
309  return result;
310  }

376  JsonMember::expected_type == JsonParseTypes::Custom ) {

379  ErrorReason::InvalidString, parse_state );
380  parse_state.remove_prefix( );

383  JsonMember::expected_type, JsonParseTypes::Real,
384  JsonParseTypes::Signed, JsonParseTypes::Unsigned,
385  JsonParseTypes::Bool, JsonParseTypes::Null> ) {
386  // All literals
387  return skip_number( parse_state );

389  JsonParseTypes::Array ) {

391  ErrorReason::InvalidArrayStart, parse_state );
392  return parse_state.skip_array( );

394  JsonParseTypes::Class ) {
395  daw_json_assert_weak( parse_state.is_opening_brace_checked( ),
396  ErrorReason::InvalidClassStart, parse_state );
397  return parse_state.skip_class( );
