119  static constexpr std::size_t max_len = 150;
120  std::size_t const len = [&]( ) -> std::size_t {
121  if( location.first == nullptr or location.class_first == nullptr ) {
122  if( location.class_first == nullptr or

124  return 0;

126  return ( std::min )(
127  static_cast<std::size_t>(
