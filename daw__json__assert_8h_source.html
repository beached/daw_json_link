110  [[maybe_unused, noreturn]] DAW_ATTRIB_NOINLINE static void
111  daw_json_error( json_details::missing_member reason,
112  ParseState const &location ) {
113 #ifdef DAW_USE_EXCEPTIONS
114  if constexpr( ShouldThrow ) {
115  using namespace std::string_literals;
116  if( location.class_first and location.first ) {
117  static constexpr std::size_t max_len = 150;
118  std::size_t const len = [&]( ) -> std::size_t {
119  if( location.first == nullptr or location.class_first == nullptr ) {
120  if( location.class_first == nullptr or
121  location.class_last == nullptr ) {
122  return 0;
123  }
124  return ( std::min )(
125  static_cast<std::size_t>(
126  std::distance( location.class_first, location.class_last ) ),
127  max_len );
128  }
129  return ( std::min )( static_cast<std::size_t>( std::distance(
130  location.class_first, location.first + 1 ) ),
131  max_len );
132  }( );
133  throw json_exception( reason,
